8 local FenUI = FenUI
9
10 local ScrollBarMixin = {}
11
12 function ScrollBarMixin:Init(config)
13 self.config = config or {}
14
15 -- Setup Slider properties
16 self:SetOrientation("VERTICAL")
17 self:SetSize(config.width or FenUI:GetLayout("scrollBarWidth"), 0) -- Height set by anchors
18
19 -- 1. Track (Background)
20 self.track = self:CreateTexture(nil, "BACKGROUND")
21 self.track:SetAllPoints()
22 local r, g, b, a = FenUI:GetColor(config.trackToken or "surfaceScrollTrack")
23 self.track:SetColorTexture(r, g, b, a)
24
25 -- 2. Thumb
26 local thumb = self:CreateTexture(nil, "ARTWORK")
27 local tr, tg, tb, ta = FenUI:GetColor(config.thumbToken or "interactiveScrollThumb")
28 thumb:SetColorTexture(tr, tg, tb, ta)
29 thumb:SetSize(config.width or FenUI:GetLayout("scrollBarWidth"), 32) -- Initial height
30 self:SetThumbTexture(thumb)
31 self.thumb = thumb
32
33 -- 3. Scripts
34 self:SetScript("OnEnter", function()
35 local hr, hg, hb, ha = FenUI:GetColor(config.thumbHoverToken or "interactiveScrollThumbHover")
36 self.thumb:SetColorTexture(hr, hg, hb, ha)
37 end)
38
39 self:SetScript("OnLeave", function()
40 local nr, ng, nb, na = FenUI:GetColor(config.thumbToken or "interactiveScrollThumb")
41 self.thumb:SetColorTexture(nr, ng, nb, na)
42 end)
43 end
44
45 function ScrollBarMixin:UpdateThumbSize(visibleHeight, totalHeight)
46 -- Handle initial/empty state
47 if totalHeight <= 0 or visibleHeight <= 0 or visibleHeight >= totalHeight then
48 if self.thumb then
49 self.thumb:SetHeight(0.1)
50 end -- Use minimal height instead of 0
51 self:Hide()
52 return
53 end
54
55 self:Show()
56 local trackHeight = self:GetHeight()
57
58 -- If track hasn't layout yet, skip update (will trigger again on resize)
59 if trackHeight <= 0 then
60 return
61 end
62
63 local ratio = visibleHeight / totalHeight
64 local thumbHeight = math.max(16, trackHeight * ratio)
65
66 self.thumb:SetHeight(thumbHeight)
67 self.thumb:Show() -- Ensure thumb is visible
68
69 -- Update slider range
70 self:SetMinMaxValues(0, totalHeight - visibleHeight)
71
72 -- Force a redraw of the thumb position
73 local current = self:GetValue()
74 self:SetValue(current)
75 end

81 function FenUI:CreateScrollBar(parent, config)
82 local scrollBar = CreateFrame("Slider", nil, parent)
83 FenUI.Mixin(scrollBar, ScrollBarMixin)
84 scrollBar:Init(config)
