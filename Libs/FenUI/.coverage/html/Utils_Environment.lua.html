13 --- Detects the client type (Retail/PTR/Beta).
14 ---@return string client "Retail", "PTR", or "Beta"
15 function Utils:GetClientType()
16 -- 1. Try native build type checks (standard WoW API globals)
17 if _G.IsBetaBuild and _G.IsBetaBuild() then
18 return "Beta"
19 end
20 if _G.IsTestBuild and _G.IsTestBuild() then
21 return "PTR"
22 end
23
24 -- 2. Fallback to portal CVar (very reliable for developers)
25 local project = GetCVar("portal") or ""
26 if project:find("test") then
27 return "PTR"
28 elseif project:find("beta") then
29 return "Beta"
30 end
31
32 -- 3. Final fallback based on interface version during transition
33 if self.IS_MIDNIGHT then
34 return "Retail"
35 end
36
37 return "Retail"
38 end

55 -- Timer font helper - maps string values to Blizzard font names
56 local fontNameMap = {
57 ["small"] = "GameFontHighlightSmallOutline",
58 ["medium"] = "GameFontHighlightOutline",
59 ["large"] = "GameFontHighlightLargeOutline",
60 ["huge"] = "GameFontHighlightHugeOutline",

63 --- Maps a size string or number to a Blizzard font name.
64 ---@param size string|number
65 ---@return string fontName
66 function Utils:GetTimerFont(size)
67 if type(size) == "string" then
68 return fontNameMap[size] or "GameFontHighlightOutline"
69 end
70 size = size or 10
71 if size <= 9 then
72 return "GameFontHighlightSmallOutline"
73 elseif size <= 12 then
74 return "GameFontHighlightOutline"
75 elseif size <= 15 then
76 return "GameFontHighlightLargeOutline"
77 else
78 return "GameFontHighlightHugeOutline"
79 end
80 end

82 -- Capability Detection (consolidated from ActionHud)
83 Utils.Cap = Utils.Cap or {}
84
85 function Utils:DetectCapabilities()
86 local Cap = self.Cap
87
88 Cap.HasSecondsFormatter = (type(_G.SecondsFormatter) ~= "nil")
89 Cap.HasHealCalculator = (type(_G.CreateUnitHealPredictionCalculator) ~= "nil")
90
91 if not self.IS_MIDNIGHT then
92 Cap.IsAuraLegacy = true
93 else
94 Cap.IsAuraLegacy = (_G.C_UnitAuras and type(_G.C_UnitAuras.GetAuraDurationRemaining) ~= "nil")
95 end
96
97 Cap.HasBooleanColor = (_G.C_CurveUtil and type(_G.C_CurveUtil.EvaluateColorFromBoolean) ~= "nil")
98 Cap.HasDurationUtil = (_G.C_DurationUtil and type(_G.C_DurationUtil.CreateDuration) ~= "nil")
99 Cap.HasSecrecyQueries = (type(_G.GetSpellAuraSecrecy) ~= "nil")
100
101 if not self.IS_MIDNIGHT then
102 Cap.IsRoyal = false
103 else
104 Cap.IsRoyal = Cap.HasSecondsFormatter or not Cap.IsAuraLegacy or Cap.HasDurationUtil or Cap.HasSecrecyQueries
105 end
106 end
