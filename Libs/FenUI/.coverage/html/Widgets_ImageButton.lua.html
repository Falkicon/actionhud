20 function ImageButtonMixin:Init(config)
21 self.config = config or {}
22
23 -- Initialize Image logic (creates texture, handles sizing/masking/conditions)
24 -- We use the exported ImageMixin:InitImage to avoid potential Init() recursion
25 self:InitImage(config)
26
27 -- Interaction states
28 self.isToggle = config.isToggle or false
29 self.isActive = config.active or false
30
31 -- Default size
32 local size = config.size or 32
33 self:SetSize(config.width or size, config.height or size)
34
35 -- Initial visuals
36 self:UpdateStateVisuals()
37
38 -- Tooltip
39 if config.tooltip then
40 self:SetScript("OnEnter", function(s)
41 s.isHovered = true
42 s:UpdateStateVisuals()
43 if s.config.onEnter then
44 s.config.onEnter(s)
45 end
46
47 GameTooltip:SetOwner(s, config.tooltipAnchor or "ANCHOR_RIGHT")
48 if type(config.tooltip) == "function" then
49 config.tooltip(GameTooltip)
50 else
51 GameTooltip:SetText(config.tooltip)
52 end
53 GameTooltip:Show()
54 end)
55 self:SetScript("OnLeave", function(s)
56 s.isHovered = false
57 s:UpdateStateVisuals()
58 if s.config.onLeave then
59 s.config.onLeave(s)
60 end
61 GameTooltip:Hide()
62 end)
63 else
64 self:SetScript("OnEnter", function(s)
65 s.isHovered = true
66 s:UpdateStateVisuals()
67 if s.config.onEnter then
68 s.config.onEnter(s)
69 end
70 end)
71 self:SetScript("OnLeave", function(s)
72 s.isHovered = false
73 s:UpdateStateVisuals()
74 if s.config.onLeave then
75 s.config.onLeave(s)
76 end
77 end)
78 end
79
80 -- Click logic
81 self:SetScript("OnMouseDown", function(s, button)
82 if button == "LeftButton" and s:IsEnabled() then
83 s.isPressed = true
84 s:UpdateStateVisuals()
85 end
86 end)
87
88 self:SetScript("OnMouseUp", function(s, button)
89 if button == "LeftButton" then
90 s.isPressed = false
91 s:UpdateStateVisuals()
92 end
93 end)
94
95 self:SetScript("OnClick", function(s, button, down)
96 if not s:IsEnabled() then
97 return
98 end
99
100 if s.isToggle and button == "LeftButton" then
101 s:SetActive(not s.isActive)
102 end
103
104 if s.config.onClick then
105 s.config.onClick(s, button, down)
106 end
107 end)
108 end

110 function ImageButtonMixin:UpdateStateVisuals()
111 if not self:IsEnabled() then
112 self:SetImageAlpha(0.3)
113 self:SetTint("interactiveDisabled")
114 return
115 end
116
117 local alpha = 1.0
118 local tint = self.config.tint or "white"
119
120 -- Priority: Pressed > Active Toggle > Hover > Normal
121 if self.isPressed then
122 alpha = 0.5 -- Darker when pressed
123 tint = self.config.pressedTint or "interactiveActive"
124 -- Slight texture shift effect for tactile feel
125 self.texture:SetPoint("TOPLEFT", 1, -1)
126 self.texture:SetPoint("BOTTOMRIGHT", 1, -1)
127 elseif self.isToggle and self.isActive then
128 alpha = 1.0
129 tint = self.config.activeTint or "interactiveActive"
130 self.texture:SetAllPoints()
131 elseif self.isHovered then
132 alpha = 1.0
133 tint = self.config.hoverTint or "interactiveHover"
134 self.texture:SetAllPoints()
135 else
136 self.texture:SetAllPoints()
137 end
138
139 self:SetImageAlpha(alpha)
140 self:SetTint(tint)
141 end

143 function ImageButtonMixin:SetActive(active)
144 self.isActive = active
145 self:UpdateStateVisuals()
146 if self.config.onToggle then
147 self.config.onToggle(self, active)
148 end
149 end
150
151 function ImageButtonMixin:GetActive()
152 return self.isActive
153 end
154
155 function ImageButtonMixin:Toggle()
156 self:SetActive(not self.isActive)
157 end
158
159 function ImageButtonMixin:SetEnabled(enabled)
160 local wasEnabled = self:IsEnabled()
161 if enabled then
162 self:Enable()
163 else
164 self:Disable()
165 end
166
167 if wasEnabled ~= enabled then
168 self:UpdateStateVisuals()
169 end
170 end

176 function FenUI:CreateImageButton(parent, config)
177 local button = CreateFrame("Button", nil, parent)
178
179 -- Order matters: ImageMixin provides texture/sizing/masking,
180 -- ImageButtonMixin provides interaction logic.
181 FenUI.Mixin(button, FenUI.ImageMixin)
182 FenUI.Mixin(button, ImageButtonMixin)
183
184 button:Init(config or {})
185 return button
186 end
