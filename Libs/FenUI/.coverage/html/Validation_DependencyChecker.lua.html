74 local function CheckAPI(apiDef)
75 local parts = { strsplit(".", apiDef.path) }
76 local current = _G
77
78 for _, part in ipairs(parts) do
79 if current == nil then
80 return false, "Parent is nil"
81 end
82 current = current[part]
83 if current == nil then
84 return false, "Not found"
85 end
86 end
87
88 return true, type(current)
89 end
90
91 local function CheckLayout(layoutName)
92 if not NineSliceLayouts then
93 return false, "NineSliceLayouts not loaded"
94 end
95
96 local layout = NineSliceLayouts[layoutName]
97 if layout then
98 return true, "exists"
99 else
100 return false, "Not found"
101 end
102 end
103
104 local function CheckAtlas(atlasName)
105 if not C_Texture or not C_Texture.GetAtlasInfo then
106 return false, "C_Texture.GetAtlasInfo not available"
107 end
108
109 local info = C_Texture.GetAtlasInfo(atlasName)
110 if info then
111 return true, string.format("%dx%d", info.width or 0, info.height or 0)
112 else
113 return false, "Not found"
114 end
115 end

121 function FenUI.Validation:Run(verbose)
122 local results = {
123 valid = true,
124 timestamp = date("%Y-%m-%d %H:%M:%S"),
125 gameVersion = GetBuildInfo(),
126 apis = { passed = 0, failed = 0, missing = {} },
127 layouts = { passed = 0, failed = 0, missing = {} },
128 atlases = { passed = 0, failed = 0, missing = {} },
129 }
130
131 -- Check APIs
132 for _, apiDef in ipairs(self.Dependencies.apis) do
133 local ok, detail = CheckAPI(apiDef)
134 if ok then
135 results.apis.passed = results.apis.passed + 1
136 if verbose then
137 print(string.format(" |cff00ff00OK|r API: %s (%s)", apiDef.name, detail))
138 end
139 else
140 results.apis.failed = results.apis.failed + 1
141 results.valid = false
142 table.insert(results.apis.missing, { name = apiDef.name, reason = detail })
143 if verbose then
144 print(string.format(" |cffff0000FAIL|r API: %s (%s)", apiDef.name, detail))
145 end
146 end
147 end
148
149 -- Check Layouts
150 for _, layoutName in ipairs(self.Dependencies.layouts) do
151 local ok, detail = CheckLayout(layoutName)
152 if ok then
153 results.layouts.passed = results.layouts.passed + 1
154 if verbose then
155 print(string.format(" |cff00ff00OK|r Layout: %s", layoutName))
156 end
157 else
158 results.layouts.failed = results.layouts.failed + 1
159 results.valid = false
160 table.insert(results.layouts.missing, { name = layoutName, reason = detail })
161 if verbose then
162 print(string.format(" |cffff0000FAIL|r Layout: %s (%s)", layoutName, detail))
163 end
164 end
165 end
166
167 -- Check Atlases
168 for _, atlasName in ipairs(self.Dependencies.atlases) do
169 local ok, detail = CheckAtlas(atlasName)
170 if ok then
171 results.atlases.passed = results.atlases.passed + 1
172 if verbose then
173 print(string.format(" |cff00ff00OK|r Atlas: %s (%s)", atlasName, detail))
174 end
175 else
176 results.atlases.failed = results.atlases.failed + 1
177 results.valid = false
178 table.insert(results.atlases.missing, { name = atlasName, reason = detail })
179 if verbose then
180 print(string.format(" |cffff0000FAIL|r Atlas: %s (%s)", atlasName, detail))
181 end
182 end
183 end
184
185 return results
186 end

192 function FenUI.Validation:PrintReport(results)
193 print(" ")
194 print("|cff88ccffFenUI Dependency Validation|r")
195 print("================================")
196 print("Game Version:", results.gameVersion)
197 print("Checked:", results.timestamp)
198 print(" ")
199
200 -- Summary
201 local apiTotal = results.apis.passed + results.apis.failed
202 local layoutTotal = results.layouts.passed + results.layouts.failed
203 local atlasTotal = results.atlases.passed + results.atlases.failed
204
205 local apiStatus = results.apis.failed == 0 and "|cff00ff00OK|r" or "|cffff6600WARN|r"
206 local layoutStatus = results.layouts.failed == 0 and "|cff00ff00OK|r" or "|cffff6600WARN|r"
207 local atlasStatus = results.atlases.failed == 0 and "|cff00ff00OK|r" or "|cffff6600WARN|r"
208
209 print(string.format("APIs: %s %d/%d", apiStatus, results.apis.passed, apiTotal))
210 print(string.format("Layouts: %s %d/%d", layoutStatus, results.layouts.passed, layoutTotal))
211 print(string.format("Atlases: %s %d/%d", atlasStatus, results.atlases.passed, atlasTotal))
212
213 -- Missing items
214 if #results.apis.missing > 0 then
215 print(" ")
216 print("|cffff6600Missing APIs:|r")
217 for _, item in ipairs(results.apis.missing) do
218 print(" -", item.name)
219 end
220 end
221
222 if #results.layouts.missing > 0 then
223 print(" ")
224 print("|cffff6600Missing Layouts:|r")
225 for _, item in ipairs(results.layouts.missing) do
226 print(" -", item.name)
227 end
228 end
229
230 if #results.atlases.missing > 0 then
231 print(" ")
232 print("|cffff6600Missing Atlases:|r")
233 for _, item in ipairs(results.atlases.missing) do
234 print(" -", item.name)
235 end
236 end
237
238 -- Final status
239 print(" ")
240 if results.valid then
241 print("|cff00ff00Status: ALL DEPENDENCIES VALID|r")
242 else
243 local totalIssues = results.apis.failed + results.layouts.failed + results.atlases.failed
244 print(string.format("|cffff0000Status: %d ISSUE(S) FOUND|r", totalIssues))
245 print("Run |cff88ccff/fenui validate verbose|r for details")
246 end
247 print(" ")
248 end

254 function FenUI.Validation:OnLoad(showWarnings)
255 -- Run silent validation
256 local results = self:Run(false)
257
258 -- Store results
259 if FenUIDB then
260 FenUIDB.lastValidation = results
261 end
262
263 -- Only warn if there are issues and warnings are enabled
264 if not results.valid and showWarnings then
265 local totalIssues = results.apis.failed + results.layouts.failed + results.atlases.failed
266 FenUI:Print(string.format("%d dependency issue(s) detected. Run /fenui validate", totalIssues))
267 end
268
269 return results
