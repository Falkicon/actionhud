42 function TabButtonMixin:UpdateVisual()
43 local r, g, b = FenUI:GetColorRGB("textDefault")
44
45 if self.isDisabled then
46 r, g, b = FenUI:GetColorRGB("interactiveDisabled")
47 if self.highlight then
48 self.highlight:Hide()
49 end
50 elseif self.isSelected then
51 r, g, b = FenUI:GetColorRGB("interactiveSelected")
52 if self.highlight then
53 self.highlight:Show()
54 end
55 elseif self.isFocused then
56 r, g, b = FenUI:GetColorRGB("interactiveHover")
57 if self.highlight then
58 self.highlight:Hide()
59 end
60 else
61 if self.highlight then
62 self.highlight:Hide()
63 end
64 end
65
66 self.text:SetTextColor(r, g, b)
67
68 -- Update badge visual
69 if self.badge then
70 local br, bg, bb =
71 FenUI:GetColorRGB(self.isDisabled and "interactiveDisabled" or (self.badgeColorToken or "feedbackSuccess"))
72 if self.badge.SetTextColor then
73 self.badge:SetTextColor(br, bg, bb)
74 elseif self.badge.SetVertexColor then
75 self.badge:SetVertexColor(br, bg, bb)
76 end
77 end
78 end

98 function TabButtonMixin:SetBadge(content, colorToken)
99 self.badgeColorToken = colorToken
100
101 local isTexture = false
102 if
103 type(content) == "string"
104 and (content:find("^atlas:") or content:find("^Interface") or content:find("%.tga$") or content:find("%.blp$"))
105 then
106 isTexture = true
107 end
108
109 -- Clean up existing badge if type mismatch
110 if self.badge then
111 local existingIsTexture = self.badge:GetObjectType() == "Texture"
112 if existingIsTexture ~= isTexture then
113 self.badge:Hide()
114 self.badge = nil
115 end
116 end
117
118 if not self.badge then
119 if isTexture then
120 self.badge = self:CreateTexture(nil, "OVERLAY")
121 self.badge:SetSize(12, 12)
122 self.badge:SetPoint("LEFT", self.text, "RIGHT", 4, 0)
123 else
124 self.badge = self:CreateFontString(nil, "OVERLAY")
125 self.badge:SetFontObject(FenUI:GetFont("fontSmall"))
126 self.badge:SetPoint("LEFT", self.text, "RIGHT", 4, 0)
127 end
128 end
129
130 if content then
131 if isTexture then
132 local texturePath = content
133 if texturePath:find("^atlas:") then
134 self.badge:SetAtlas(texturePath:sub(7))
135 else
136 self.badge:SetTexture(texturePath)
137 end
138 else
139 self.badge:SetText(content)
140 end
141 self.badge:Show()
142 else
143 self.badge:Hide()
144 end
145
146 self:UpdateVisual()
147 self:UpdateWidth()
148 end
149
150 function TabButtonMixin:GetBadge()
151 return self.badge and self.badge:GetText()

160 function TabGroupMixin:Init(config)
161 self.config = config or {}
162 self.tabs = {}
163 self.tabOrder = {}
164 self.selectedKey = nil
165 self.hooks = {
166 onChange = config.onChange,
167 onTabCreate = config.onTabCreate,
168 onTabFocus = config.onTabFocus,
169 }
170
171 -- Create tabs from config
172 if config.tabs then
173 for _, tabDef in ipairs(config.tabs) do
174 local key = tabDef.key or tabDef.id
175 local tab = self:AddTab(key, tabDef.text, tabDef.icon)
176 if tabDef.disabled then
177 tab:SetDisabled(true)
178 end
179 end
180 end
181
182 -- Select first tab by default
183 if #self.tabOrder > 0 and not self.selectedKey then
184 -- Find first non-disabled tab
185 for _, key in ipairs(self.tabOrder) do
186 if not self.tabs[key].isDisabled then
187 self:Select(key)
188 break
189 end
190 end
191 end
192 end
193
194 function TabGroupMixin:AddTab(key, text, icon)
195 if self.tabs[key] then
196 FenUI:Debug("Tab already exists:", key)
197 return self.tabs[key]
198 end
199
200 local tab = CreateFrame("Button", nil, self)
201 FenUI.Mixin(tab, TabButtonMixin)
202
203 tab.key = key
204
205 -- Create text
206 tab.text = tab:CreateFontString(nil, "OVERLAY")
207 tab.text:SetFontObject(FenUI:GetFont("fontButton"))
208 tab.text:SetPoint("CENTER")
209
210 -- Create highlight/underline
211 tab.highlight = tab:CreateTexture(nil, "HIGHLIGHT")
212 tab.highlight:SetColorTexture(FenUI:GetColorRGB("interactiveSelected"))
213 tab.highlight:SetHeight(2)
214
215 -- Position highlight based on group position
216 if self.config.position == "bottom" then
217 tab.highlight:SetPoint("TOPLEFT", 4, 0)
218 tab.highlight:SetPoint("TOPRIGHT", -4, 0)
219 else
220 tab.highlight:SetPoint("BOTTOMLEFT", 4, 0)
221 tab.highlight:SetPoint("BOTTOMRIGHT", -4, 0)
222 end
223 tab.highlight:Hide()
224
225 -- Scripts
226 tab:SetScript("OnClick", function()
227 self:Select(key)
228 end)
229
230 tab:SetScript("OnEnter", function(btn)
231 if not btn.isDisabled and not btn.isSelected then
232 local r, g, b = FenUI:GetColorRGB("interactiveHover")
233 btn.text:SetTextColor(r, g, b)
234 end
235 end)
236
237 tab:SetScript("OnLeave", function(btn)
238 btn:UpdateVisual()
239 end)
240
241 -- Focus support
242 tab:SetScript("OnReceiveDrag", function()
243 self:SetFocus(key)
244 end)
245
246 -- Set text and size
247 tab:SetTabText(text)
248 tab:SetHeight(self.config.height or 28)
249
250 -- Store
251 self.tabs[key] = tab
252 table.insert(self.tabOrder, key)
253
254 -- Update layout
255 self:RepositionTabs()
256
257 -- Fire hook
258 if self.hooks.onTabCreate then
259 self.hooks.onTabCreate(tab, key)
260 end
261
262 return tab
263 end

291 function TabGroupMixin:RepositionTabs()
292 local prevTab = nil
293 local spacing = FenUI:GetSpacing("spacingElement")
294
295 for _, key in ipairs(self.tabOrder) do
296 local tab = self.tabs[key]
297 if tab:IsShown() then
298 tab:ClearAllPoints()
299 if prevTab then
300 tab:SetPoint("LEFT", prevTab, "RIGHT", spacing, 0)
301 else
302 tab:SetPoint("LEFT", 0, 0)
303 end
304 prevTab = tab
305 end
306 end
307 end

309 function TabGroupMixin:Select(key)
310 local tab = self.tabs[key]
311 if not tab or tab.isDisabled then
312 return
313 end
314
315 local previousKey = self.selectedKey
316 self.selectedKey = key
317
318 -- Update all tabs
319 for k, t in pairs(self.tabs) do
320 t:SetSelected(k == key)
321 end
322
323 -- Fire callback
324 if self.hooks.onChange and key ~= previousKey then
325 self.hooks.onChange(key, previousKey)
326 end
327 end

360 --- Create a tab group
361 ---@param parent Frame Parent frame
362 ---@param config table Configuration { tabs, position, height, onChange, etc }
363 ---@return Frame tabGroup
364 function FenUI:CreateTabGroup(parent, config)
365 config = config or {}
366
367 local tabGroup = CreateFrame("Frame", config.name, parent)
368 FenUI.Mixin(tabGroup, TabGroupMixin)
369
370 tabGroup:SetHeight(config.height or 32)
371 if config.width then
372 tabGroup:SetWidth(config.width)
373 end
374
375 tabGroup:Init(config)
376
377 return tabGroup
378 end
