10 function WidgetMixin:Init(config)
11 self.config = config or {}
12 self.items = config.items or {}
13
14 -- Main Button
15 local button = FenUI:CreateButton(self, {
16 text = self.config.defaultText or "Select...",
17 width = self:GetWidth(),
18 height = self:GetHeight(),

28 function WidgetMixin:UpdateMenuList()
29 self.menuList = {}
30 local U = FenUI.Utils
31
32 for _, item in ipairs(self.items) do
33 local isTable = type(item) == "table"
34 -- Use explicit nil check to avoid returning the table itself if text is missing
35 local text = (isTable and item.text ~= nil) and item.text or item
36 local value = (isTable and item.value ~= nil) and item.value or item
37
38 -- Fallback for tables missing text but having value
39 if text == item and isTable and item.value ~= nil then
40 text = tostring(item.value)
41 end
42
43 -- Sanitize text for Blizzard SetText (handles AceLocale 'true')
44 text = U and U:SanitizeText(text, tostring(value)) or (text == true and tostring(value) or text)
45
46 table.insert(self.menuList, {
47 text = text,
48 func = function()
49 if not self.config.fixedText then
50 self.button:SetText(text)
51 end
52 if self.config.onSelect then
53 self.config.onSelect(value)
54 end
55 end,
56 notCheckable = true,
57 })
58 end
59 end

61 function WidgetMixin:ToggleMenu()
62 if not self.items or #self.items == 0 then
63 return
64 end
65
66 if not self.menuList then
67 self:UpdateMenuList()
68 end
69
70 -- Use the shared ShowMenu utility (bridged or direct)
71 if FenUI.Utils and FenUI.Utils.ShowMenu then
72 FenUI.Utils:ShowMenu(self.menuList, self.button)
73 end
74 end

81 function WidgetMixin:SetValue(value)
82 local U = FenUI.Utils
83 for _, item in ipairs(self.items) do
84 local isTable = type(item) == "table"
85 local itemValue = (isTable and item.value ~= nil) and item.value or item
86 local itemText = (isTable and item.text ~= nil) and item.text or item
87
88 if itemValue == value then
89 -- Fallback for tables missing text but having value
90 if itemText == item and isTable and item.value ~= nil then
91 itemText = tostring(itemValue)
92 end
93
94 -- Sanitize text for Blizzard SetText (handles AceLocale 'true')
95 itemText = U and U:SanitizeText(itemText, tostring(itemValue))
96 or (itemText == true and tostring(itemValue) or itemText)
97 self.button:SetText(itemText)
98 return
99 end
100 end
101 end
