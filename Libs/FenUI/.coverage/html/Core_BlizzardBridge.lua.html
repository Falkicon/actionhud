107 local BORDER_PIECES = {
108 "TopLeftCorner",
109 "TopRightCorner",
110 "BottomLeftCorner",
111 "BottomRightCorner",
112 "TopEdge",
113 "BottomEdge",
114 "LeftEdge",
115 "RightEdge",
116 }

123 end
124
125 --- Apply a custom 8-texture border to a frame
126 ---@param frame Frame The target frame
127 ---@param borderKey string The border pack key from Tokens.lua
128 ---@param colorToken string|nil Optional color token to tint the border
129 ---@param margin table|nil Optional margin {top, bottom, left, right}
130 function FenUI:ApplyBorder(frame, borderKey, colorToken, margin)
131 local pack = self:GetBorderPack(borderKey)
132 if not pack then
133 FenUI:Debug("Border pack not found:", borderKey)
134 return false
135 end
136
137 local m = margin or { top = 0, bottom = 0, left = 0, right = 0 }
138
139 -- 1. Create or clear existing border textures
140 frame.customBorder = frame.customBorder or {}
141 local pieces = frame.customBorder
142
143 -- Ensure we have all 8 pieces
144 for _, name in ipairs(BORDER_PIECES) do
145 if not pieces[name] then
146 pieces[name] = frame:CreateTexture(nil, "BORDER", nil, 5)
147 end
148 local tex = pieces[name]
149 tex:SetTexture(pack.file)
150 tex:Show()
151 end
152
153 -- 2. Setup TexCoords (Slicing)
154 -- The texture is assumed to be a grid where corners are 'slice' pixels square
155 -- and edges are 1px thick between corners.
156 -- We use standard 0-1 normalized coordinates.
157 -- Note: This implementation assumes a square texture atlas for simplicity.
158 local s = pack.slice / 64 -- Standardizing on 64px source textures for now
159
160 pieces.TopLeftCorner:SetTexCoord(0, s, 0, s)
161 pieces.TopRightCorner:SetTexCoord(1 - s, 1, 0, s)
162 pieces.BottomLeftCorner:SetTexCoord(0, s, 1 - s, 1)
163 pieces.BottomRightCorner:SetTexCoord(1 - s, 1, 1 - s, 1)
164
165 pieces.TopEdge:SetTexCoord(s, 1 - s, 0, s)
166 pieces.BottomEdge:SetTexCoord(s, 1 - s, 1 - s, 1)
167 pieces.LeftEdge:SetTexCoord(0, s, s, 1 - s)
168 pieces.RightEdge:SetTexCoord(1 - s, 1, s, 1 - s)
169
170 -- 3. Positioning
171 local size = pack.slice
172 pieces.TopLeftCorner:SetSize(size, size)
173 pieces.TopLeftCorner:SetPoint("TOPLEFT", m.left, -m.top)
174
175 pieces.TopRightCorner:SetSize(size, size)
176 pieces.TopRightCorner:SetPoint("TOPRIGHT", -m.right, -m.top)
177
178 pieces.BottomLeftCorner:SetSize(size, size)
179 pieces.BottomLeftCorner:SetPoint("BOTTOMLEFT", m.left, m.bottom)
180
181 pieces.BottomRightCorner:SetSize(size, size)
182 pieces.BottomRightCorner:SetPoint("BOTTOMRIGHT", -m.right, m.bottom)
183
184 pieces.TopEdge:SetPoint("TOPLEFT", pieces.TopLeftCorner, "TOPRIGHT")
185 pieces.TopEdge:SetPoint("TOPRIGHT", pieces.TopRightCorner, "TOPLEFT")
186 pieces.TopEdge:SetHeight(size)
187
188 pieces.BottomEdge:SetPoint("BOTTOMLEFT", pieces.BottomLeftCorner, "BOTTOMRIGHT")
189 pieces.BottomEdge:SetPoint("BOTTOMRIGHT", pieces.BottomRightCorner, "BOTTOMLEFT")
190 pieces.BottomEdge:SetHeight(size)
191
192 pieces.LeftEdge:SetPoint("TOPLEFT", pieces.TopLeftCorner, "BOTTOMLEFT")
193 pieces.LeftEdge:SetPoint("BOTTOMLEFT", pieces.BottomLeftCorner, "TOPLEFT")
194 pieces.LeftEdge:SetWidth(size)
195
196 pieces.RightEdge:SetPoint("TOPRIGHT", pieces.TopRightCorner, "BOTTOMRIGHT")
197 pieces.RightEdge:SetPoint("BOTTOMRIGHT", pieces.BottomRightCorner, "TOPRIGHT")
198 pieces.RightEdge:SetWidth(size)
199
200 -- 4. Theming (Coloring)
201 local r, g, b, a = self:GetColor(colorToken or "borderDefault")
202 for _, tex in pairs(pieces) do
203 tex:SetVertexColor(r, g, b, a)
204 end
205
206 -- Store state
207 frame.borderApplied = true
208 frame.fenUIBorderKey = borderKey
209
210 return true
211 end

225 -- NineSlice Application
226 --------------------------------------------------------------------------------
227
228 --- Apply a NineSlice layout to a frame
229 ---@param frame Frame The frame to apply the layout to
230 ---@param layoutName string The layout name (FenUI alias or Blizzard name)
231 ---@param textureKit string|nil Optional texture kit for themed layouts
232 ---@param margin table|nil Optional margin {top, bottom, left, right}
233 ---@return boolean success
234 function FenUI:ApplyLayout(frame, layoutName, textureKit, margin)
235 if not NineSliceUtil or not NineSliceLayouts then
236 FenUI:Debug("NineSliceUtil not available")
237 return false
238 end
239
240 local blizzardLayoutName = self:ResolveLayoutName(layoutName)
241 local layout = NineSliceLayouts[blizzardLayoutName]
242
243 if not layout then
244 FenUI:Debug("Layout not found:", blizzardLayoutName)
245 return false
246 end
247
248 -- Apply the layout
249 NineSliceUtil.ApplyLayout(frame, layout, textureKit)
250
251 -- Apply margins if provided
252 if margin then
253 local m = margin
254 if frame.TopLeftCorner then
255 frame.TopLeftCorner:ClearAllPoints()
256 frame.TopLeftCorner:SetPoint("TOPLEFT", m.left, -m.top)
257 end
258 if frame.TopRightCorner then
259 frame.TopRightCorner:ClearAllPoints()
260 frame.TopRightCorner:SetPoint("TOPRIGHT", -m.right, -m.top)
261 end
262 if frame.BottomLeftCorner then
263 frame.BottomLeftCorner:ClearAllPoints()
264 frame.BottomLeftCorner:SetPoint("BOTTOMLEFT", m.left, m.bottom)
265 end
266 if frame.BottomRightCorner then
267 frame.BottomRightCorner:ClearAllPoints()
268 frame.BottomRightCorner:SetPoint("BOTTOMRIGHT", -m.right, m.bottom)
269 end
270 -- Note: Edges are usually anchored to corners by NineSliceUtil,
271 -- so they should follow automatically.
272 end
273
274 -- Store layout info on the frame
275 frame.fenUILayout = blizzardLayoutName
276 frame.fenUITextureKit = textureKit
277
278 FenUI:Debug("Applied layout:", blizzardLayoutName, textureKit and ("with kit: " .. textureKit) or "")
279 return true
280 end

282 --- Apply a layout by Blizzard name directly (bypasses alias lookup)
283 ---@param frame Frame The frame to apply the layout to
284 ---@param blizzardLayoutName string The exact Blizzard layout name
285 ---@param textureKit string|nil Optional texture kit
286 ---@return boolean success
287 function FenUI:ApplyLayoutDirect(frame, blizzardLayoutName, textureKit)
288 if not NineSliceUtil or not NineSliceLayouts then
289 return false
290 end
291
292 local layout = NineSliceLayouts[blizzardLayoutName]
293 if not layout then
294 return false
295 end
296
297 NineSliceUtil.ApplyLayout(frame, layout, textureKit)
298 frame.fenUILayout = blizzardLayoutName
299 frame.fenUITextureKit = textureKit
300
301 return true
302 end

308 --- Get the NineSlice pieces from a frame (if it has them)
309 ---@param frame Frame The frame with a NineSlice layout
310 ---@return table|nil pieces Table of piece names -> textures
311 function FenUI:GetLayoutPieces(frame)
312 local pieceNames = {
313 "TopLeftCorner",
314 "TopRightCorner",
315 "BottomLeftCorner",
316 "BottomRightCorner",
317 "TopEdge",
318 "BottomEdge",
319 "LeftEdge",
320 "RightEdge",
321 "Center",
322 }
323
324 local pieces = {}
325 local found = false
326
327 for _, name in ipairs(pieceNames) do
328 if frame[name] then
329 pieces[name] = frame[name]
330 found = true
331 end
332 end
333
334 return found and pieces or nil
335 end
