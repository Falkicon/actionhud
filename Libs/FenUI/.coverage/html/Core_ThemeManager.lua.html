34 --- Register a new theme
35 ---@param name string Theme identifier
36 ---@param config table Theme configuration
37 function FenUI.ThemeManager:Register(name, config)
38 if FenUI.Themes[name] then
39 FenUI:Debug("Overwriting existing theme:", name)
40 end
41
42 FenUI.Themes[name] = {
43 name = config.name or name,
44 description = config.description or "",
45 textureKit = config.textureKit,
46 layout = config.layout or "Panel",
47 tokens = config.tokens or {},
48 }
49
50 FenUI:Debug("Registered theme:", name)
51 end

53 --- Unregister a theme
54 ---@param name string Theme identifier
55 function FenUI.ThemeManager:Unregister(name)
56 if name ~= "Default" then
57 FenUI.Themes[name] = nil
58 FenUI:Debug("Unregistered theme:", name)
59 end
60 end

109 --- Apply a theme to a specific frame
110 ---@param frame Frame The frame to theme
111 ---@param themeName string Theme identifier
112 ---@return boolean success
113 function FenUI.ThemeManager:ApplyToFrame(frame, themeName)
114 local theme = FenUI.Themes[themeName]
115 if not theme then
116 FenUI:Debug("Theme not found:", themeName)
117 return false
118 end
119
120 -- Apply the border (custom pack or native layout)
121 if theme.layout and frame.fenUISupportsLayout and frame.SetBorder then
122 frame:SetBorder(theme.layout)
123 end
124
125 -- Apply Blizzard textureKit if using legacy NineSlice
126 if theme.textureKit and frame.fenUISupportsLayout and frame.fenUILayout then
127 FenUI:ApplyLayout(frame, frame.fenUILayout, theme.textureKit)
128 end
129
130 -- Store theme info on frame
131 frame.fenUITheme = themeName
132
133 -- Update frame registry
134 if frame.fenUIFrameId and FenUI.registeredFrames[frame.fenUIFrameId] then
135 FenUI.registeredFrames[frame.fenUIFrameId].theme = themeName
136 end
137
138 -- Call frame's theme update method if it exists
139 if frame.OnFenUIThemeChanged then
140 frame:OnFenUIThemeChanged(themeName, theme)
141 end
142
143 FenUI:Debug("Applied theme", themeName, "to frame")
144 return true
145 end

147 --- Set the global theme (updates all registered frames)
148 ---@param themeName string Theme identifier
149 ---@return boolean success
150 function FenUI.ThemeManager:SetGlobalTheme(themeName)
151 local theme = FenUI.Themes[themeName]
152 if not theme then
153 FenUI:Print("Theme not found:", themeName)
154 return false
155 end
156
157 -- Update global state
158 FenUI.currentGlobalTheme = themeName
159 if FenUIDB then
160 FenUIDB.globalTheme = themeName
161 end
162
163 -- Apply token overrides
164 FenUI:ApplyTokenOverrides(theme.tokens)
165
166 -- Update all registered frames
167 local updatedCount = 0
168 for frameId, data in pairs(FenUI.registeredFrames) do
169 if data.frame and data.frame:IsObjectType("Frame") then
170 self:ApplyToFrame(data.frame, themeName)
171 updatedCount = updatedCount + 1
172 end
173 end
174
175 -- Fire callbacks
176 for _, callback in pairs(FenUI.themeChangeCallbacks) do
177 local ok, err = pcall(callback, themeName)
178 if not ok then
179 FenUI:Debug("Theme callback error:", err)
180 end
181 end
182
183 FenUI:Print("Global theme set to:", theme.name, "(" .. updatedCount .. " frames updated)")
184 return true
185 end
