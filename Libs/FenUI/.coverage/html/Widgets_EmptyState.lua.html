2 -- FenUI v2 - EmptyState Widget
3 --
4 -- Centered overlay for empty content areas with two-slot architecture.
5 --
6 -- Slots:
7 -- - top: Typically an Image, icon, or custom frame
8 -- - bottom: Typically text (title + subtitle), or custom composite
9 --
10 -- Features:
11 -- - Background (color, gradient, or image)
12 -- - Slot-based content injection
13 -- - Backwards compatible with icon/title/subtitle props
14 -- - Creates Image component for conditional icons

25 function EmptyStateMixin:Init(config)
26 self.config = config or {}
27 self.slots = {}
28
29 -- 1. Background
30 self.bg = self:CreateTexture(nil, "BACKGROUND")
31 self.bg:SetAllPoints()
32
33 if config.backgroundGradient then
34 local g = config.backgroundGradient
35 local fromR, fromG, fromB, fromA = FenUI:GetColor(g.from or "black")
36 local toR, toG, toB, toA = FenUI:GetColor(g.to or "transparent")
37 self.bg:SetGradient(
38 g.direction or "VERTICAL",
39 CreateColor(fromR, fromG, fromB, fromA),
40 CreateColor(toR, toG, toB, toA)
41 )
42 elseif config.backgroundImage then
43 self.bg:SetTexture(config.backgroundImage)
44 elseif config.background then
45 local r, g, b, a = FenUI:GetColor(config.background)
46 self.bg:SetColorTexture(r, g, b, a)
47 else
48 self.bg:Hide()
49 end
50
51 -- 2. Content Container (centered)
52 self.content = CreateFrame("Frame", nil, self)
53 self.content:SetPoint("CENTER")
54 local contentWidth = config.width or (self:GetParent() and self:GetParent():GetWidth()) or 300
55 self.content:SetSize(contentWidth, 200)
56
57 -- 3. Create slots container frames
58 self.topSlot = CreateFrame("Frame", nil, self.content)
59 self.topSlot:SetPoint("TOP", self.content, "TOP", 0, 0)
60
61 self.bottomSlot = CreateFrame("Frame", nil, self.content)
62
63 -- Gap between slots
64 local gap = config.gap or 16
65 if type(gap) == "string" then
66 gap = FenUI:GetSpacing(gap) or 16
67 end
68 self.slotGap = gap
69
70 -- 4. Populate slots based on config (backwards compatibility)
71 self:PopulateFromConfig(config)
72
73 -- 5. Layout slots
74 self:LayoutSlots()
75 end

77 --- Populate slots from config (backwards compatibility layer)
78 function EmptyStateMixin:PopulateFromConfig(config)
79 -- TOP SLOT: Image or icon
80 if config.image then
81 -- New image config - create Image component
82 if FenUI.CreateImage then
83 local imageConfig = type(config.image) == "table" and config.image or { texture = config.image }
84 -- Apply default size if not specified (default reduced to 32)
85 imageConfig.width = imageConfig.width or config.iconSize or 32
86 imageConfig.height = imageConfig.height or imageConfig.width
87
88 local img = FenUI:CreateImage(self.topSlot, imageConfig)
89 img:SetPoint("CENTER")
90 self.topSlot:SetSize(imageConfig.width, imageConfig.height)
91 self.slots.top = img
92 end
93 elseif config.icon then
94 -- Legacy icon prop - create simple texture or Image component (default reduced to 32)
95 local iconSize = config.iconSize or 32
96
97 if FenUI.CreateImage then
98 local img = FenUI:CreateImage(self.topSlot, {
99 texture = config.icon,
100 width = iconSize,
101 height = iconSize,
102 })
103 img:SetPoint("CENTER")
104 self.slots.top = img
105 else
106 -- Fallback to raw texture
107 local icon = self.topSlot:CreateTexture(nil, "ARTWORK")
108 icon:SetTexture(config.icon)
109 icon:SetSize(iconSize, iconSize)
110 icon:SetPoint("CENTER")
111 self.slots.top = icon
112 end
113
114 self.topSlot:SetSize(iconSize, iconSize)
115 end
116
117 -- BOTTOM SLOT: Title and/or subtitle
118 if config.title or config.subtitle then
119 local textHeight = 0
120
121 -- Title
122 if config.title then
123 self.titleText = self.bottomSlot:CreateFontString(nil, "OVERLAY")
124 self.titleText:SetFontObject(FenUI:GetFont("fontHeading"))
125 self.titleText:SetText(config.title)
126 self.titleText:SetPoint("TOP", self.bottomSlot, "TOP", 0, 0)
127 self.titleText:SetJustifyH("CENTER")
128
129 local r, g, b = FenUI:GetColorRGB(config.titleToken or "textEmptyTitle")
130 self.titleText:SetTextColor(r, g, b)
131
132 textHeight = textHeight + self.titleText:GetStringHeight()
133 end
134
135 -- Subtitle
136 if config.subtitle then
137 self.subtitleText = self.bottomSlot:CreateFontString(nil, "OVERLAY")
138 self.subtitleText:SetFontObject(FenUI:GetFont("fontSmall"))
139 self.subtitleText:SetText(config.subtitle)
140 self.subtitleText:SetJustifyH("CENTER")
141 self.subtitleText:SetWidth(self.content:GetWidth() - 20)
142
143 local r, g, b = FenUI:GetColorRGB(config.subtitleToken or "textEmptySubtitle")
144 self.subtitleText:SetTextColor(r, g, b)
145
146 if self.titleText then
147 self.subtitleText:SetPoint("TOP", self.titleText, "BOTTOM", 0, -4)
148 else
149 self.subtitleText:SetPoint("TOP", self.bottomSlot, "TOP", 0, 0)
150 end
151
152 textHeight = textHeight + self.subtitleText:GetStringHeight() + 4
153 end
154
155 self.bottomSlot:SetSize(self.content:GetWidth(), textHeight)
156 end
157 end

159 --- Layout slots vertically with gap
160 function EmptyStateMixin:LayoutSlots()
161 local totalHeight = 0
162 local topHeight = self.topSlot:GetHeight() or 0
163 local bottomHeight = self.bottomSlot:GetHeight() or 0
164
165 -- Calculate total content height
166 if topHeight > 0 then
167 totalHeight = totalHeight + topHeight
168 end
169 if bottomHeight > 0 then
170 if topHeight > 0 then
171 totalHeight = totalHeight + self.slotGap
172 end
173 totalHeight = totalHeight + bottomHeight
174 end
175
176 -- Update content container height
177 self.content:SetHeight(math.max(totalHeight, 50))
178
179 -- Position slots
180 self.topSlot:ClearAllPoints()
181 self.topSlot:SetPoint("TOP", self.content, "TOP", 0, 0)
182
183 self.bottomSlot:ClearAllPoints()
184 if topHeight > 0 then
185 self.bottomSlot:SetPoint("TOP", self.topSlot, "BOTTOM", 0, -self.slotGap)
186 else
187 self.bottomSlot:SetPoint("TOP", self.content, "TOP", 0, 0)
188 end
189 end

191 --- Set content for a slot
192 ---@param slotName string "top" or "bottom"
193 ---@param frame Frame|nil Frame to place in slot, or nil to clear
194 function EmptyStateMixin:SetSlot(slotName, frame)
195 local slotFrame = slotName == "top" and self.topSlot or self.bottomSlot
196
197 -- Clear existing content
198 self:ClearSlot(slotName)
199
200 if frame then
201 frame:SetParent(slotFrame)
202 frame:SetPoint("CENTER")
203 self.slots[slotName] = frame
204
205 -- Update slot size to match frame
206 local width, height = frame:GetSize()
207 if width > 0 and height > 0 then
208 slotFrame:SetSize(width, height)
209 end
210 end
211
212 self:LayoutSlots()
213 end

215 --- Clear a slot's content
216 ---@param slotName string "top" or "bottom"
217 function EmptyStateMixin:ClearSlot(slotName)
218 local existing = self.slots[slotName]
219 if existing then
220 if existing.Hide then
221 existing:Hide()
222 end
223 self.slots[slotName] = nil
224 end
225
226 local slotFrame = slotName == "top" and self.topSlot or self.bottomSlot
227 slotFrame:SetSize(0, 0)
228 end

273 --- Create an empty state component
274 ---@param parent Frame Parent frame
275 ---@param config table Configuration { background, image, icon, title, subtitle, gap, etc }
276 ---@return Frame emptyState
277 function FenUI:CreateEmptyState(parent, config)
278 local empty = CreateFrame("Frame", nil, parent)
279 empty:SetAllPoints()
280 FenUI.Mixin(empty, EmptyStateMixin)
281 empty:Init(config or {})
