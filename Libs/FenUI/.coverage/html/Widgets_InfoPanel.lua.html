13 function InfoPanelMixin:InitInfoPanel(config)
14 config = config or {}
15 self.infoConfig = config
16 self.sectionFrames = {}
17
18 -- Default padding for InfoPanel if not specified (more breathing room)
19 local padding = config.padding or 12
20 self:SetPadding(padding)
21
22 -- Use the Panel's content area
23 local content = self:GetContentFrame()
24
25 -- Create scroll panel for content
26 local scrollBarWidth = FenUI:GetLayout("scrollBarWidth") or 20
27 local scrollPanel = FenUI:CreateScrollPanel(content, {
28 padding = 0,
29 showScrollBar = true,
30 })
31 scrollPanel:SetAllPoints()
32 self.scrollFrame = scrollPanel.scrollFrame
33 self.scrollChild = scrollPanel.scrollChild
34
35 -- Adjust scroll panel if close button is shown
36 if config.showCloseButton then
37 scrollPanel:SetPoint("BOTTOMRIGHT", content, "BOTTOMRIGHT", 0, 40)
38 end
39
40 -- Update scroll child width and layout sections on size change
41 self.scrollFrame:SetScript("OnSizeChanged", function(sf, width)
42 self.scrollChild:SetWidth(width)
43 self:LayoutSections()
44 end)
45
46 -- Close button at bottom is now optional and disabled by default
47 if config.showCloseButton then
48 local closeBtn = FenUI:CreateButton(self.safeZone, {
49 text = "Close",
50 width = 100,
51 onClick = function()
52 self:Hide()
53 end,
54 })
55 closeBtn:SetPoint("BOTTOM", 0, 8)
56 self.closeBtn = closeBtn
57
58 -- Adjust scroll frame to leave room for the button if it's shown
59 self.scrollFrame:SetPoint("BOTTOMRIGHT", -scrollBarWidth, 40)
60 end
61
62 -- Initialize sections if provided
63 if config.sections then
64 self:SetSections(config.sections)
65 end
66 end

68 function InfoPanelMixin:SetSections(sections)
69 -- Clear existing sections
70 for _, sectionFrame in ipairs(self.sectionFrames) do
71 sectionFrame:Hide()
72 sectionFrame:ClearAllPoints()
73 end
74 wipe(self.sectionFrames)
75
76 -- Create new sections
77 for i, sectionData in ipairs(sections) do
78 local section = FenUI:CreateSection(self.scrollChild, {
79 heading = sectionData.heading,
80 body = sectionData.body,
81 headingColor = sectionData.headingColor,
82 bodyColor = sectionData.bodyColor,
83 })
84 section:SetPoint("LEFT", 0, 0)
85 section:SetPoint("RIGHT", 0, 0)
86 self.sectionFrames[i] = section
87 end
88
89 self:LayoutSections()
90 end
91
92 function InfoPanelMixin:LayoutSections()
93 local yOffset = 0
94 local sectionGap = FenUI:GetSpacing("md") -- 16px between sections
95
96 for i, section in ipairs(self.sectionFrames) do
97 section:ClearAllPoints()
98 section:SetPoint("TOPLEFT", self.scrollChild, "TOPLEFT", 0, -yOffset)
99 section:SetPoint("TOPRIGHT", self.scrollChild, "TOPRIGHT", 0, -yOffset)
100
101 -- Trigger height recalculation
102 section:UpdateHeight()
103 local sectionHeight = section:GetHeight()
104
105 yOffset = yOffset + sectionHeight + sectionGap
106 end
107
108 -- Set scroll child height
109 self.scrollChild:SetHeight(math.max(1, yOffset))
110 end
111
112 function InfoPanelMixin:AddSection(heading, body)
113 local section = FenUI:CreateSection(self.scrollChild, {
114 heading = heading,
115 body = body,
116 })
117 section:SetPoint("LEFT", 0, 0)
118 section:SetPoint("RIGHT", 0, 0)
119 table.insert(self.sectionFrames, section)
120 self:LayoutSections()
121 end
122
123 function InfoPanelMixin:ClearSections()
124 for _, sectionFrame in ipairs(self.sectionFrames) do
125 sectionFrame:Hide()
126 sectionFrame:ClearAllPoints()
127 end
128 wipe(self.sectionFrames)
129 self.scrollChild:SetHeight(1)
130 end
131
132 function InfoPanelMixin:ScrollToTop()
133 if self.scrollFrame then
134 self.scrollFrame:SetVerticalScroll(0)
135 end
136 end

142 --- Create an info panel for displaying structured content
143 ---@param parent Frame Parent frame
144 ---@param config table { title, width, height, sections, movable, closable }
145 ---@return Frame infoPanel
146 function FenUI:CreateInfoPanel(parent, config)
147 config = config or {}
148
149 -- Set defaults
150 config.width = config.width or 450
151 config.height = config.height or 400
152 config.movable = config.movable ~= false
153 config.closable = config.closable ~= false
154
155 -- Create base panel
156 local panel = FenUI:CreatePanel(parent or UIParent, config)
157 panel:SetFrameStrata("DIALOG")
158 panel:SetPoint("CENTER")
159
160 -- Apply InfoPanel mixin
161 FenUI.Mixin(panel, InfoPanelMixin)
162 panel:InitInfoPanel(config)
163
164 -- Hide by default (caller shows when ready)
165 panel:Hide()
166
167 return panel
168 end

170 FenUI.InfoPanelMixin = InfoPanelMixin
