36 function StackMixin:Init(config)
37 self.config = config or {}
38 self.children = {}
39
40 -- Core properties
41 self.direction = config.direction or "vertical"
42 self.gap = config.gap or 0
43 self.align = config.align or "start"
44 self.justify = config.justify or "start"
45 self.wrap = config.wrap or false
46 self.rowGap = config.rowGap or self.gap
47
48 -- Initialize Layout layers (inherited via factory)
49 if self.InitLayout then
50 self:InitLayout(config)
51 end
52
53 -- Hook for layout updates
54 -- We use a wrapper to ensure we don't break Layout's own OnSizeChanged
55 -- which is responsible for updating background anchors.
56 self:HookScript("OnSizeChanged", function()
57 self:Layout()
58 end)
59
60 -- If children were provided in config, add them
61 if config.children then
62 for _, child in ipairs(config.children) do
63 if type(child) == "table" and child.frame then
64 self:AddChild(child.frame, child.config)
65 else
66 self:AddChild(child)
67 end
68 end
69 end
70 end

72 --- Add a child to the stack
73 ---@param frame Frame The child frame
74 ---@param config table|nil Optional child-level overrides (align, grow, shrink)
75 function StackMixin:AddChild(frame, config)
76 if not frame then
77 return
78 end
79
80 local childData = {
81 frame = frame,
82 config = config or {},
83 baseWidth = frame:GetWidth(),
84 baseHeight = frame:GetHeight(),
85 }
86
87 table.insert(self.children, childData)
88
89 -- Ensure parent is set directly to self (the stack container)
90 -- This avoids coordinate system confusion with Layout's internal contentFrame.
91 frame:SetParent(self)
92
93 -- Auto-show children when added to a stack
94 frame:Show()
95
96 -- Hook child size and visibility changes
97 if not frame.stackHooked then
98 frame:HookScript("OnSizeChanged", function()
99 if self.Layout then
100 self:Layout()
101 end
102 end)
103 frame:HookScript("OnShow", function()
104 if self.Layout then
105 self:Layout()
106 end
107 end)
108 frame:HookScript("OnHide", function()
109 if self.Layout then
110 self:Layout()
111 end
112 end)
113 frame.stackHooked = true
114 end
115
116 self:Layout()
117 end

137 --- Calculate positions and sizes for all children
138 function StackMixin:Layout()
139 if self.isLayouting or #self.children == 0 then
140 return
141 end
142 self.isLayouting = true
143
144 local isVertical = self.direction == "vertical"
145 local gap = self:ResolveGap(self.gap)
146 local rowGap = self:ResolveGap(self.rowGap)
147 local padding = self.GetPadding and self:GetPadding() or { top = 0, bottom = 0, left = 0, right = 0 }
148
149 -- Use config values as primary source of truth for size
150 local initialWidth = tonumber(self.config.width)
151 or (self.dynamicSize and tonumber(self.dynamicSize.width))
152 or self:GetWidth()
153 local initialHeight = tonumber(self.config.height)
154 or (self.dynamicSize and tonumber(self.dynamicSize.height))
155 or self:GetHeight()
156
157 local totalWidth = initialWidth - (padding.left + padding.right)
158 local totalHeight = initialHeight - (padding.top + padding.bottom)
159
160 -- Measure Phase: Collect visible children and their sizes
161 local childrenToLayout = {}
162 local maxCrossSize = 0
163 local totalMainSize = 0
164 local totalGrow = 0
165 local needsDeferredLayout = false
166
167 for _, childData in ipairs(self.children) do
168 local frame = childData.frame
169 if frame:IsShown() then
170 local w, h = childData.baseWidth, childData.baseHeight
171
172 -- If base size is 0, try to get current size as fallback
173 if w <= 0 or h <= 0 then
174 local fw, fh = frame:GetSize()
175 if w <= 0 then w = fw end
176 if h <= 0 then h = fh end
177 end
178
179 -- If still 0 size, try fallbacks
180 if w <= 0 then
181 if frame.config and frame.config.width and tonumber(frame.config.width) then
182 w = tonumber(frame.config.width)
183 elseif frame.dynamicSize and frame.dynamicSize.width and tonumber(frame.dynamicSize.width) then
184 w = tonumber(frame.dynamicSize.width)
185 elseif frame:IsObjectType("Button") then
186 w = 100
187 end
188 end
189
190 if h <= 0 then
191 if frame.config and frame.config.height and tonumber(frame.config.height) then
192 h = tonumber(frame.config.height)
193 elseif frame.dynamicSize and frame.dynamicSize.height and tonumber(frame.dynamicSize.height) then
194 h = tonumber(frame.dynamicSize.height)
195 elseif frame:IsObjectType("Button") then
196 h = 24
197 end
198 end
199
200 local grow = childData.config.grow or 0
201 totalGrow = totalGrow + grow
202
203 -- If we still have 0 size and it's not a grow item, we might need a defer
204 if (w <= 0 or h <= 0) and (grow == 0) then
205 needsDeferredLayout = true
206 end
207
208 table.insert(childrenToLayout, {
209 frame = frame,
210 config = childData.config,
211 width = w,
212 height = h,
213 grow = grow,
214 })
215
216 if isVertical then
217 maxCrossSize = math.max(maxCrossSize, w)
218 totalMainSize = totalMainSize + h
219 else
220 maxCrossSize = math.max(maxCrossSize, h)
221 totalMainSize = totalMainSize + w
222 end
223 end
224 end
225
226 if needsDeferredLayout and not self.hasDeferredOnce then
227 self.hasDeferredOnce = true
228 C_Timer.After(0, function()
229 self.isLayouting = nil
230 self:Layout()
231 end)
232 self.isLayouting = nil
233 return
234 end
235
236 local numChildren = #childrenToLayout
237 if numChildren == 0 then
238 self.isLayouting = nil
239 return
240 end
241
242 totalMainSize = totalMainSize + (gap * (numChildren - 1))
243
244 -- Auto-sizing support
245 local isAutoWidth = self.config.width == "auto" or (not self.config.width and initialWidth <= 0)
246 local isAutoHeight = self.config.height == "auto" or (not self.config.height and initialHeight <= 0)
247
248 if isVertical then
249 if isAutoHeight then
250 totalHeight = totalMainSize
251 self:SetHeight(totalHeight + padding.top + padding.bottom)
252 end
253 if isAutoWidth then
254 totalWidth = maxCrossSize
255 self:SetWidth(totalWidth + padding.left + padding.right)
256 end
257 else
258 if isAutoWidth then
259 totalWidth = totalMainSize
260 self:SetWidth(totalWidth + padding.left + padding.right)
261 end
262 if isAutoHeight then
263 totalHeight = maxCrossSize
264 self:SetHeight(totalHeight + padding.top + padding.bottom)
265 end
266 end
267
268 if totalWidth <= 0 or totalHeight <= 0 then
269 self.isLayouting = nil
270 return
271 end
272
273 -- Debug info for Mechanic
274 self.debug_totalWidth = totalWidth
275 self.debug_totalHeight = totalHeight
276 self.debug_numChildren = numChildren
277 self.debug_totalGrow = totalGrow
278
279 if self.wrap then
280 self:LayoutWrapped(childrenToLayout, totalWidth, totalHeight, padding, gap, rowGap, isVertical)
281 else
282 self:LayoutStandard(childrenToLayout, totalWidth, totalHeight, padding, gap, isVertical, totalGrow, totalMainSize)
283 end
284
285 self.isLayouting = nil
286 end

288 function StackMixin:LayoutStandard(
289 childrenToLayout,
290 totalWidth,
291 totalHeight,
292 padding,
293 gap,
294 isVertical,
295 totalGrow,
296 totalFixedSize
297 )
298 local numChildren = #childrenToLayout
299 local availableSize = (isVertical and totalHeight or totalWidth)
300 local remainingSize = availableSize - totalFixedSize
301 local flexUnit = totalGrow > 0 and math.max(0, remainingSize / totalGrow) or 0
302
303 -- Justification Setup
304 local startOffset = 0
305 local justifiedGap = gap
306
307 if totalGrow == 0 and remainingSize > 0 then
308 if self.justify == "center" then
309 startOffset = remainingSize / 2
310 elseif self.justify == "end" then
311 startOffset = remainingSize
312 elseif self.justify == "space-between" and numChildren > 1 then
313 justifiedGap = gap + (remainingSize / (numChildren - 1))
314 elseif self.justify == "space-around" then
315 justifiedGap = gap + (remainingSize / numChildren)
316 startOffset = (justifiedGap - gap) / 2
317 end
318 end
319
320 local currentOffset = startOffset
321 for _, layoutData in ipairs(childrenToLayout) do
322 local frame = layoutData.frame
323 local childConfig = layoutData.config
324 local align = childConfig.align or self.align
325
326 local childWidth = layoutData.width
327 local childHeight = layoutData.height
328
329 if layoutData.grow > 0 then
330 local growSize = flexUnit * layoutData.grow
331 if isVertical then
332 childHeight = childHeight + growSize
333 else
334 childWidth = childWidth + growSize
335 end
336 end
337
338 frame:ClearAllPoints()
339
340 -- Ensure child is above container background
341 frame:SetFrameLevel(self:GetFrameLevel() + 2)
342
343 if isVertical then
344 if align == "start" then
345 frame:SetPoint("TOPLEFT", self, "TOPLEFT", padding.left, -(padding.top + currentOffset))
346 elseif align == "center" then
347 frame:SetPoint("TOP", self, "TOP", (padding.left - padding.right) / 2, -(padding.top + currentOffset))
348 elseif align == "end" then
349 frame:SetPoint("TOPRIGHT", self, "TOPRIGHT", -padding.right, -(padding.top + currentOffset))
350 elseif align == "stretch" then
351 frame:SetPoint("TOPLEFT", self, "TOPLEFT", padding.left, -(padding.top + currentOffset))
352 childWidth = totalWidth
353 end
354
355 frame:SetWidth(math.max(1, childWidth))
356 frame:SetHeight(math.max(1, childHeight))
357
358 -- Debug info
359 self["debug_child" .. _ .. "_pos"] = string.format("%.1f, %.1f (%.1f x %.1f)", padding.left, -(padding.top + currentOffset), childWidth, childHeight)
360
361 currentOffset = currentOffset + childHeight + justifiedGap
362 else
363 if align == "start" then
364 frame:SetPoint("TOPLEFT", self, "TOPLEFT", padding.left + currentOffset, -padding.top)
365 elseif align == "center" then
366 frame:SetPoint("LEFT", self, "LEFT", padding.left + currentOffset, (padding.bottom - padding.top) / 2)
367 elseif align == "end" then
368 frame:SetPoint("BOTTOMLEFT", self, "BOTTOMLEFT", padding.left + currentOffset, padding.bottom)
369 elseif align == "stretch" then
370 frame:SetPoint("TOPLEFT", self, "TOPLEFT", padding.left + currentOffset, -padding.top)
371 childHeight = totalHeight
372 end
373
374 frame:SetWidth(math.max(1, childWidth))
375 frame:SetHeight(math.max(1, childHeight))
376
377 -- Debug info
378 self["debug_child" .. _ .. "_pos"] = string.format("%.1f, %.1f (%.1f x %.1f)", padding.left + currentOffset, -padding.top, childWidth, childHeight)
379
380 currentOffset = currentOffset + childWidth + justifiedGap
381 end
382 end
383 end

385 function StackMixin:LayoutWrapped(childrenToLayout, totalWidth, totalHeight, padding, gap, rowGap, isVertical)
386 local lines = {}
387 local currentLine = { children = {}, size = 0, maxSize = 0 }
388 table.insert(lines, currentLine)
389
390 local mainMaxSize = isVertical and totalHeight or totalWidth
391
392 -- Group into lines
393 for _, layoutData in ipairs(childrenToLayout) do
394 local childSize = isVertical and layoutData.height or layoutData.width
395 local childCrossSize = isVertical and layoutData.width or layoutData.height
396
397 if #currentLine.children > 0 and currentLine.size + gap + childSize > mainMaxSize then
398 currentLine = { children = {}, size = 0, maxSize = 0 }
399 table.insert(lines, currentLine)
400 end
401
402 if #currentLine.children > 0 then
403 currentLine.size = currentLine.size + gap
404 end
405
406 table.insert(currentLine.children, layoutData)
407 currentLine.size = currentLine.size + childSize
408 currentLine.maxSize = math.max(currentLine.maxSize, childCrossSize)
409 end
410
411 -- Position lines and children
412 local crossOffset = 0
413 for i, line in ipairs(lines) do
414 local mainOffset = 0
415 local numInLine = #line.children
416 local remainingInLine = mainMaxSize - line.size
417
418 -- Justification within line
419 local startOffset = 0
420 local justifiedGap = gap
421
422 if remainingInLine > 0 then
423 if self.justify == "center" then
424 startOffset = remainingInLine / 2
425 elseif self.justify == "end" then
426 startOffset = remainingInLine
427 elseif self.justify == "space-between" and numInLine > 1 then
428 justifiedGap = gap + (remainingInLine / (numInLine - 1))
429 elseif self.justify == "space-around" then
430 justifiedGap = gap + (remainingInLine / numInLine)
431 startOffset = (justifiedGap - gap) / 2
432 end
433 end
434
435 mainOffset = startOffset
436
437 for _, layoutData in ipairs(line.children) do
438 local frame = layoutData.frame
439 local align = layoutData.config.align or self.align
440 local childWidth = layoutData.width
441 local childHeight = layoutData.height
442
443 frame:ClearAllPoints()
444
445 if isVertical then
446 -- Vertical Wrap: Columns
447 if align == "start" then
448 frame:SetPoint("TOPLEFT", self, "TOPLEFT", padding.left + crossOffset, -(padding.top + mainOffset))
449 elseif align == "center" then
450 frame:SetPoint(
451 "TOPLEFT",
452 self,
453 "TOPLEFT",
454 padding.left + crossOffset + (line.maxSize - childWidth) / 2,
455 -(padding.top + mainOffset)
456 )
457 elseif align == "end" then
458 frame:SetPoint(
459 "TOPLEFT",
460 self,
461 "TOPLEFT",
462 padding.left + crossOffset + (line.maxSize - childWidth),
463 -(padding.top + mainOffset)
464 )
465 elseif align == "stretch" then
466 frame:SetPoint("TOPLEFT", self, "TOPLEFT", padding.left + crossOffset, -(padding.top + mainOffset))
467 childWidth = line.maxSize
468 end
469
470 frame:SetWidth(math.max(1, childWidth))
471 frame:SetHeight(math.max(1, childHeight))
472 mainOffset = mainOffset + childHeight + justifiedGap
473 else
474 -- Horizontal Wrap: Rows
475 if align == "start" then
476 frame:SetPoint("TOPLEFT", self, "TOPLEFT", padding.left + mainOffset, -(padding.top + crossOffset))
477 elseif align == "center" then
478 frame:SetPoint(
479 "TOPLEFT",
480 self,
481 "TOPLEFT",
482 padding.left + mainOffset,
483 -(padding.top + crossOffset + (line.maxSize - childHeight) / 2)
484 )
485 elseif align == "end" then
486 frame:SetPoint(
487 "TOPLEFT",
488 self,
489 "TOPLEFT",
490 padding.left + mainOffset,
491 -(padding.top + crossOffset + (line.maxSize - childHeight))
492 )
493 elseif align == "stretch" then
494 frame:SetPoint("TOPLEFT", self, "TOPLEFT", padding.left + mainOffset, -(padding.top + crossOffset))
495 childHeight = line.maxSize
496 end
497
498 frame:SetWidth(math.max(1, childWidth))
499 frame:SetHeight(math.max(1, childHeight))
500 mainOffset = mainOffset + childWidth + justifiedGap
501 end
502 end
503
504 crossOffset = crossOffset + line.maxSize + rowGap
505 end
506 end

508 function StackMixin:ResolveGap(gapValue)
509 if type(gapValue) == "string" then
510 local val = FenUI:GetSpacing(gapValue)
511 if val == 0 then
512 val = FenUI:GetLayout(gapValue)
513 end
514 return val
515 elseif type(gapValue) == "number" then
516 return gapValue
517 elseif type(gapValue) == "table" then
518 return gapValue.row or gapValue.gap or 0
519 end
520 return 0
521 end

531 function FenUI:CreateStack(parent, config)
532 config = config or {}
533
534 -- Inherit from Layout for border/background support
535 local stack = self:CreateLayout(parent, config)
536
537 -- Mix in Stack properties
538 stack.InitLayout = stack.Init -- Save Layout:Init
539 FenUI.Mixin(stack, StackMixin)
540
541 -- Re-initialize as Stack
542 stack:Init(config)
543
544 return stack
545 end
