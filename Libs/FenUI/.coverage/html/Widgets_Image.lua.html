40 -- Calculate texture coords and size for sizing modes
41 -- Returns: width, height, texCoords (or nil for full texture)
42 local function CalculateSizing(mode, frameWidth, frameHeight, texWidth, texHeight)
43 if not texWidth or texWidth == 0 or not texHeight or texHeight == 0 then
44 return frameWidth, frameHeight, nil
45 end
46
47 local frameAspect = frameWidth / frameHeight
48 local texAspect = texWidth / texHeight
49
50 if mode == "fill" then
51 -- Stretch to fill exactly (no aspect preservation)
52 return frameWidth, frameHeight, nil
53 elseif mode == "fit" then
54 -- Scale to fit within bounds, preserve aspect, may have gaps
55 if texAspect > frameAspect then
56 -- Texture is wider, fit to width
57 local h = frameWidth / texAspect
58 return frameWidth, h, nil
59 else
60 -- Texture is taller, fit to height
61 local w = frameHeight * texAspect
62 return w, frameHeight, nil
63 end
64 elseif mode == "contain" then
65 -- Same as fit - scale to fit, preserve aspect
66 if texAspect > frameAspect then
67 local h = frameWidth / texAspect
68 return frameWidth, h, nil
69 else
70 local w = frameHeight * texAspect
71 return w, frameHeight, nil
72 end
73 elseif mode == "cover" then
74 -- Scale to cover bounds, preserve aspect, crop if needed
75 if texAspect > frameAspect then
76 -- Texture is wider, fit to height and crop width
77 local scale = frameHeight / texHeight
78 local scaledWidth = texWidth * scale
79 local cropX = (scaledWidth - frameWidth) / 2 / scaledWidth
80 return frameWidth, frameHeight, { cropX, 0, 1 - cropX, 1 }
81 else
82 -- Texture is taller, fit to width and crop height
83 local scale = frameWidth / texWidth
84 local scaledHeight = texHeight * scale
85 local cropY = (scaledHeight - frameHeight) / 2 / scaledHeight
86 return frameWidth, frameHeight, { 0, cropY, 1, 1 - cropY }
87 end
88 end
89
90 -- Default: fill
91 return frameWidth, frameHeight, nil
92 end

113 function ImageMixin:InitImage(config)
114 self.config = config or {}
115
116 -- Create main texture
117 if not self.texture then
118 self.texture = self:CreateTexture(nil, config.drawLayer or "ARTWORK")
119 self.texture:SetAllPoints()
120
121 -- Improve sharpness for small icons
122 if self.texture.SetFilterMode then
123 self.texture:SetFilterMode("TRILINEAR")
124 end
125 if self.texture.SetTexelSnappingBias then
126 self.texture:SetTexelSnappingBias(0.0)
127 end
128 if self.texture.SetSnapToPixelGrid then
129 self.texture:SetSnapToPixelGrid(true)
130 end
131 end
132
133 -- Frame sizing - fill mode vs explicit size
134 if config.fill then
135 -- Fill mode: anchor to all parent edges
136 self:ClearAllPoints()
137 self:SetPoint("TOPLEFT")
138 self:SetPoint("BOTTOMRIGHT")
139 self.isFillMode = true
140 else
141 -- Explicit size mode
142 local size = config.size or 64
143 local width = config.width or size
144 local height = config.height or width
145 self:SetSize(width, height)
146 self.isFillMode = false
147 end
148
149 -- Store sizing mode
150 self.sizingMode = config.sizing or "fill"
151
152 -- Apply mask if specified
153 if config.mask then
154 self:SetMask(config.mask)
155 end
156
157 -- Interactive setup
158 if config.onClick or config.onEnter or config.onLeave or config.tooltip then
159 self:EnableMouse(true)
160
161 if config.onClick then
162 self:SetScript("OnMouseUp", function(_, button)
163 if button == "LeftButton" then
164 config.onClick(self)
165 end
166 end)
167 end
168
169 self:SetScript("OnEnter", function()
170 if config.onEnter then
171 config.onEnter(self)
172 end
173 if config.tooltip then
174 GameTooltip:SetOwner(self, "ANCHOR_RIGHT")
175 GameTooltip:SetText(config.tooltip)
176 GameTooltip:Show()
177 end
178 end)
179
180 self:SetScript("OnLeave", function()
181 if config.onLeave then
182 config.onLeave(self)
183 end
184 if config.tooltip then
185 GameTooltip:Hide()
186 end
187 end)
188 end
189
190 -- Apply alpha
191 if config.alpha then
192 self.texture:SetAlpha(config.alpha)
193 end
194
195 -- Resolve and apply texture
196 self:Refresh()
197
198 -- Apply tint after texture is set
199 if config.tint then
200 self:SetTint(config.tint)
201 end
202 end

204 --- Resolve the texture path based on condition or direct config
205 ---@return string|nil texturePath
206 function ImageMixin:ResolveTexture()
207 local config = self.config
208
209 -- Conditional resolution
210 if config.condition or config.resolver then
211 local resolver = config.resolver
212
213 -- Use built-in resolver if condition name provided
214 if config.condition and CONDITION_RESOLVERS[config.condition] then
215 resolver = CONDITION_RESOLVERS[config.condition]
216 end
217
218 if resolver and config.variants then
219 local key = resolver()
220 local texture = config.variants[key]
221
222 if texture then
223 return texture, false -- not atlas
224 end
225 end
226
227 -- Fallback
228 return config.fallback, false
229 end
230
231 -- Direct texture
232 if config.texture then
233 return config.texture, false
234 end
235
236 -- Atlas texture
237 if config.atlas then
238 return config.atlas, true
239 end
240
241 -- Fallback
242 return config.fallback, false
243 end

245 --- Refresh the image (re-resolve conditional and apply)
246 function ImageMixin:Refresh()
247 local texturePath, isAtlas = self:ResolveTexture()
248
249 if not texturePath then
250 -- No texture, hide
251 self.texture:Hide()
252 return
253 end
254
255 self.texture:Show()
256
257 if isAtlas then
258 self.texture:SetAtlas(texturePath)
259 else
260 self.texture:SetTexture(texturePath)
261 end
262
263 -- Apply sizing mode
264 self:ApplySizing()
265 end

274 --- Apply sizing mode calculations
275 function ImageMixin:ApplySizing()
276 local frameWidth, frameHeight = self:GetSize()
277
278 -- For now, we can't easily get texture dimensions in WoW
279 -- So sizing modes are simplified:
280 -- - fill: stretch to frame size
281 -- - fit/contain: assume square texture, scale proportionally
282 -- - cover: fill and potentially use SetTexCoord for cropping
283
284 local mode = self.sizingMode
285
286 if mode == "fill" then
287 self.texture:SetAllPoints()
288 self.texture:SetTexCoord(0, 1, 0, 1)
289 elseif mode == "fit" or mode == "contain" then
290 -- For custom aspect ratio handling, would need texture dimension callback
291 -- Default to fill behavior
292 self.texture:SetAllPoints()
293 self.texture:SetTexCoord(0, 1, 0, 1)
294 elseif mode == "cover" then
295 self.texture:SetAllPoints()
296 self.texture:SetTexCoord(0, 1, 0, 1)
297 end
298 end

331 --- Apply tinting to the image
332 ---@param tokenOrColor string|table Token name or {r, g, b, a} table
333 function ImageMixin:SetTint(tokenOrColor)
334 if type(tokenOrColor) == "string" then
335 -- Token name
336 local r, g, b, a = FenUI:GetColor(tokenOrColor)
337 self.texture:SetVertexColor(r, g, b, a)
338 elseif type(tokenOrColor) == "table" then
339 -- Direct color
340 self.texture:SetVertexColor(
341 tokenOrColor[1] or 1,
342 tokenOrColor[2] or 1,
343 tokenOrColor[3] or 1,
344 tokenOrColor[4] or 1
345 )
346 end
347 end
348
349 --- Clear tinting (reset to white)
350 function ImageMixin:ClearTint()

352 end

354 --- Apply a mask to the image
355 ---@param maskType string "circle", "rounded", or texture path
356 function ImageMixin:SetMask(maskType)
357 local maskTexture = MASK_TEXTURES[maskType] or maskType
358
359 if maskTexture and maskTexture ~= "Interface\\BUTTONS\\WHITE8X8" then
360 -- Create mask frame if needed
361 if not self.maskTexture then
362 self.maskTexture = self:CreateMaskTexture()
363 self.maskTexture:SetAllPoints()
364 end
365
366 self.maskTexture:SetTexture(maskTexture)
367 self.texture:AddMaskTexture(self.maskTexture)
368 end
369 end

372 function ImageMixin:ClearMask()
