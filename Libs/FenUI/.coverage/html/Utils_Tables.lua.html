17 copy[self:DeepCopy(orig_key)] = self:DeepCopy(orig_value)
18 end

26 --- Safe comparison that handles secret values in Midnight.
27 --- Returns nil if comparison is not possible.
28 ---@param a any
29 ---@param b any
30 ---@param op string? "<", ">", "<=", ">=", "==", "~="
31 ---@return boolean|nil result
32 function Utils:SafeCompare(a, b, op)
33 op = op or "=="
34 local ok, result = pcall(function()
35 if a == nil or b == nil then
36 return a == b
37 end
38
39 local aSecret = self:IsValueSecret(a)
40 local bSecret = self:IsValueSecret(b)
41
42 if aSecret or bSecret then
43 -- In Midnight, secrets cannot be compared with operators.
44 -- We only allow equality checks if they are the exact same object.
45 if op == "==" then
46 return rawequal(a, b)
47 elseif op == "~=" then
48 return not rawequal(a, b)
49 end
50 return nil -- Comparison not possible
51 end
52
53 if type(a) == "table" and type(b) == "table" and op == "==" then
54 -- Simple deep compare for table values
55 for k, v in pairs(a) do
56 if not self:SafeCompare(v, b[k], "==") then
57 return false
58 end
59 end
60 for k, v in pairs(b) do
61 if a[k] == nil then
62 return false
63 end
64 end
65 return true
66 end
67
68 if op == ">" then
69 return a > b
70 elseif op == "<" then
71 return a < b
72 elseif op == ">=" then
73 return a >= b
74 elseif op == "<=" then
75 return a <= b
76 elseif op == "==" then
77 return a == b
78 elseif op == "~=" then
79 return a ~= b
80 end
81 return a == b
82 end)
83
84 if ok then
85 return result
86 else
87 return nil
88 end
89 end
