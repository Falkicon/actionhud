166 function GridMixin:UpdateLayout()
167 local width = self:GetWidth()
168 local colDefs = self.config.columns
169 local gap = type(self.config.gap) == "table" and self.config.gap.column or self.config.gap or 0
170 local totalGap = gap * (#colDefs - 1)
171 local availableWidth = width - totalGap
172
173 -- Calculate column widths
174 local colWidths = {}
175 local totalFr = 0
176 local usedWidth = 0
177
178 -- Pass 1: Fixed and Auto
179 for i, def in ipairs(colDefs) do
180 if type(def) == "number" then
181 colWidths[i] = def
182 usedWidth = usedWidth + def
183 elseif type(def) == "string" and def:find("px$") then
184 local val = tonumber(def:match("^(%d+)"))
185 colWidths[i] = val
186 usedWidth = usedWidth + val
187 elseif def == "auto" then
188 -- For simplicity in WoW, auto behaves like 1fr for now unless content measured
189 -- True CSS auto is hard without pre-measuring all content
190 totalFr = totalFr + 1
191 elseif type(def) == "string" and def:find("fr$") then
192 local val = tonumber(def:match("^(%d+)")) or 1
193 totalFr = totalFr + val
194 end
195 end
196
197 -- Pass 2: Fractional
198 local frWidth = totalFr > 0 and (availableWidth - usedWidth) / totalFr or 0
199 for i, def in ipairs(colDefs) do
200 if not colWidths[i] then
201 local fr = 1
202 if type(def) == "string" and def:find("fr$") then
203 fr = tonumber(def:match("^(%d+)")) or 1
204 end
205 colWidths[i] = fr * frWidth
206 end
207 end
208
209 -- Apply to all visible rows
210 for _, row in ipairs(self.rows) do
211 self:ApplyRowLayout(row, colWidths, gap)
212 end
213
214 self.colWidths = colWidths
215 self.colGap = gap
216 end
217
218 function GridMixin:ApplyRowLayout(row, widths, gap)
219 local xOffset = 0
220 for i, cellWidth in ipairs(widths) do
221 local cell = row:GetCell(i)
222 cell:ClearAllPoints()
223 cell:SetPoint("TOPLEFT", row, "TOPLEFT", xOffset, 0)
224 cell:SetPoint("BOTTOMLEFT", row, "BOTTOMLEFT", xOffset, 0)
225 cell:SetWidth(math.max(1, cellWidth))
226 xOffset = xOffset + cellWidth + gap
227 end
228 end

230 function GridMixin:AddRow()
231 local row = table.remove(self.rowPool)
232 if not row then
233 row = CreateFrame("Button", nil, self)
234 FenUI.Mixin(row, GridRowMixin)
235 row:Init(self, #self.rows + 1)
236 end
237
238 local rowHeight = self.config.rowHeight
239 row:SetHeight(rowHeight)
240
241 -- Position
242 local prevRow = self.rows[#self.rows]
243 local rowGap = type(self.config.gap) == "table" and self.config.gap.row or self.config.gap or 0
244
245 row:ClearAllPoints()
246 if prevRow then
247 row:SetPoint("TOPLEFT", prevRow, "BOTTOMLEFT", 0, -rowGap)
248 row:SetPoint("TOPRIGHT", prevRow, "BOTTOMRIGHT", 0, -rowGap)
249 else
250 row:SetPoint("TOPLEFT", 0, 0)
251 row:SetPoint("TOPRIGHT", 0, 0)
252 end
253
254 table.insert(self.rows, row)
255
256 if self.colWidths then
257 self:ApplyRowLayout(row, self.colWidths, self.colGap)
258 end
259
260 row:Show()
261 return row
262 end
263
264 function GridMixin:Clear()
265 for _, row in ipairs(self.rows) do
266 row:Hide()
267 table.insert(self.rowPool, row)
268 end
269 wipe(self.rows)
270 end

282 function GridMixin:Refresh()
283 self:Clear()
284
285 local rowGap = type(self.config.gap) == "table" and self.config.gap.row or self.config.gap or 0
286 local totalHeight = 0
287
288 for i, item in ipairs(self.data) do
289 local row = self:AddRow()
290 row:Bind(item, i)
291
292 if self.hooks.onRowBind then
293 self.hooks.onRowBind(row, item, i)
294 end
295
296 totalHeight = totalHeight + self.config.rowHeight + (i > 1 and rowGap or 0)
297 end
298
299 self:SetHeight(math.max(1, totalHeight))
300 end
