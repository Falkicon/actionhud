93 --- Get margin values (consistent with Layout principle)
94 ---@return table { top, bottom, left, right }
95 function GroupMixin:GetMargin()
96 local config = self.config
97 local margin = config.margin
98 local base = { top = 0, bottom = 0, left = 0, right = 0 }
99
100 if margin then
101 if type(margin) == "number" then
102 base = { top = margin, bottom = margin, left = margin, right = margin }
103 elseif type(margin) == "string" then
104 local val = FenUI:GetSpacing(margin)
105 if val == 0 then
106 val = FenUI:GetLayout(margin)
107 end
108 base = { top = val, bottom = val, left = val, right = val }
109 elseif type(margin) == "table" then
110 base = {
111 top = margin.top or 0,
112 bottom = margin.bottom or 0,
113 left = margin.left or 0,
114 right = margin.right or 0,
115 }
116 end
117 end
118
119 -- Apply individual side overrides
120 if config.marginTop then
121 base.top = FenUI:GetSpacing(config.marginTop)
122 end
123 if config.marginBottom then
124 base.bottom = FenUI:GetSpacing(config.marginBottom)
125 end
126 if config.marginLeft then
127 base.left = FenUI:GetSpacing(config.marginLeft)
128 end
129 if config.marginRight then
130 base.right = FenUI:GetSpacing(config.marginRight)
131 end
132
133 return base
134 end

179 --- Get the content frame (creates if needed for single-cell mode)
180 ---@return Frame
181 function GroupMixin:GetContentFrame()
182 if not self.contentFrame then
183 self.contentFrame = CreateFrame("Frame", nil, self)
184
185 local p = self:GetPadding()
186 local m = self:GetMargin()
187 self.contentFrame:SetPoint("TOPLEFT", m.left + p.left, -(m.top + p.top))
188
189 if self.dynamicSize and (self.dynamicSize.width == "auto" or self.dynamicSize.height == "auto") then
190 -- Let children define the size
191 else
192 self.contentFrame:SetPoint("BOTTOMRIGHT", -(m.right + p.right), m.bottom + p.bottom)
193 end
194 end
195 return self.contentFrame
196 end

216 function GroupMixin:CreateCells()
217 local rowDefs = self.config.rows
218 local colDefs = self.config.cols
219 local defs = rowDefs or colDefs
220
221 if not defs then
222 return
223 end
224
225 -- Handle numeric defs by creating an array of "1fr"
226 if type(defs) == "number" then
227 local num = defs
228 defs = {}
229 for i = 1, num do
230 table.insert(defs, "1fr")
231 end
232 end
233
234 if #defs == 0 then
235 return
236 end
237
238 self.orientation = rowDefs and "VERTICAL" or "HORIZONTAL"
239 local parsedDefs = {}
240
241 for i, def in ipairs(defs) do
242 if def == "auto" then
243 parsedDefs[i] = { type = "auto", value = 0 }
244 elseif type(def) == "number" then
245 parsedDefs[i] = { type = "fixed", value = def }
246 elseif type(def) == "string" and def:find("px$") then
247 local val = tonumber(def:match("^(%d+)")) or 0
248 parsedDefs[i] = { type = "fixed", value = val }
249 elseif type(def) == "string" and def:find("fr$") then
250 local val = tonumber(def:match("^(%d+)")) or 1
251 parsedDefs[i] = { type = "fr", value = val }
252 else
253 parsedDefs[i] = { type = "fr", value = 1 }
254 end
255 end
256
257 for i = 1, #defs do
258 local cell = CreateFrame("Frame", nil, self)
259 cell.index = i
260 cell.def = parsedDefs[i]
261 self.cells[i] = cell
262 end
263
264 self:SetScript("OnSizeChanged", function()
265 self:LayoutCells()
266 end)
267
268 self:LayoutCells()
269 end

271 function GroupMixin:LayoutCells()
272 if #self.cells == 0 then
273 return
274 end
275
276 local p = self:GetPadding()
277 local m = self:GetMargin()
278 local gap = self:ResolveGap()
279 local isVertical = self.orientation == "VERTICAL"
280
281 local totalSize = isVertical and (self:GetHeight() - (p.top + p.bottom + m.top + m.bottom))
282 or (self:GetWidth() - (p.left + p.right + m.left + m.right))
283
284 local totalGaps = gap * (#self.cells - 1)
285 local availableSize = math.max(0, totalSize - totalGaps)
286
287 local fixedSize = 0
288 local totalFr = 0
289
290 for _, cell in ipairs(self.cells) do
291 if cell.def.type == "fixed" then
292 fixedSize = fixedSize + cell.def.value
293 elseif cell.def.type == "fr" then
294 totalFr = totalFr + cell.def.value
295 end
296 end
297
298 local frSize = totalFr > 0 and math.max(0, availableSize - fixedSize) / totalFr or 0
299
300 local offset = isVertical and (p.top + m.top) or (p.left + m.left)
301 for i, cell in ipairs(self.cells) do
302 local cellSize
303 if cell.def.type == "fixed" then
304 cellSize = cell.def.value
305 elseif cell.def.type == "fr" then
306 cellSize = frSize * cell.def.value
307 else
308 cellSize = frSize
309 end
310
311 cell:ClearAllPoints()
312 if isVertical then
313 cell:SetPoint("TOPLEFT", p.left + m.left, -offset)
314 cell:SetPoint("TOPRIGHT", -(p.right + m.right), -offset)
315 cell:SetHeight(math.max(1, cellSize))
316 else
317 cell:SetPoint("TOPLEFT", offset, -(p.top + m.top))
318 cell:SetPoint("BOTTOMLEFT", offset, p.bottom + m.bottom)
319 cell:SetWidth(math.max(1, cellSize))
320 end
321
322 offset = offset + cellSize + gap
323 end
324 end

330 function GroupMixin:ResolveGap()
331 local gap = self.config.gap
332 if type(gap) == "string" then
333 return FenUI:GetSpacing(gap)
334 elseif type(gap) == "number" then
335 return gap
336 elseif type(gap) == "table" then
337 return gap.row or 0
338 end
339 return 0
340 end

346 --- Create a lightweight structural group
347 ---@param parent Frame Parent frame
348 ---@param config table Configuration { name, width, height, padding, gap, rows, cols }
349 ---@return Frame group
350 function FenUI:CreateGroup(parent, config)
351 config = config or {}
352
353 -- Ensure predictable naming for inspector UI
354 local frameName = config.name and ("FenUIGroup_" .. config.name) or nil
355 local group = CreateFrame("Frame", frameName, parent or UIParent)
356
357 FenUI.Mixin(group, GroupMixin)
358 group:Init(config)
359
360 return group
361 end
