16 function MultiLineEditBoxMixin:Init(config)
17 self.config = config or {}
18
19 -- Create scroll panel
20 local scrollBarWidth = FenUI:GetLayout("scrollBarWidth") or 20
21 self.scrollPanel = FenUI:CreateScrollPanel(self, {
22 padding = 4,
23 showScrollBar = true,
24 })
25 self.scrollPanel:SetAllPoints()
26 self.scrollFrame = self.scrollPanel.scrollFrame
27
28 -- Create edit box
29 self.editBox = CreateFrame("EditBox", nil, self.scrollFrame)
30 self.editBox:SetMultiLine(true)
31 self.editBox:SetMaxLetters(0)
32
33 local fontToken = config.font or "fontBody"
34 local fontObject = FenUI:GetFont(fontToken) or "ChatFontNormal"
35 self.editBox:SetFontObject(fontObject)
36
37 self.editBox:SetWidth(self.scrollFrame:GetWidth())
38 self.editBox:SetAutoFocus(false)
39
40 self.scrollFrame:SetScrollChild(self.editBox)
41
42 -- Update ScrollPanel child reference
43 self.scrollPanel.scrollChild = self.editBox
44
45 -- Create hidden measurement font string
46 self.measureFS = self:CreateFontString(nil, "ARTWORK")
47 self.measureFS:Hide()
48 local font, size, flags = self.editBox:GetFont()
49 if font then
50 self.measureFS:SetFont(font, size, flags)
51 end
52 self.measureFS:SetWidth(self.scrollFrame:GetWidth())
53
54 -- Helper to get text height
55 function self:GetTextHeight()
56 self.measureFS:SetText(self.editBox:GetText() or "")
57 return self.measureFS:GetHeight()
58 end
59
60 -- Configure behavior
61 if config.readOnly then
62 self:SetReadOnly(true)
63 end
64
65 if config.label then
66 self:SetLabel(config.label)
67 end
68
69 -- Initial text
70 if config.text then
71 self:SetText(config.text)
72 end
73
74 -- Handle size changes
75 self:HookScript("OnSizeChanged", function(_, width, height)
76 local editBoxWidth = width - 30
77 self.editBox:SetWidth(editBoxWidth)
78 self.measureFS:SetWidth(editBoxWidth)
79
80 -- Ensure editBox is at least as tall as the scroll frame so it's clickable
81 -- and can handle text selection in empty space.
82 if height and height > 8 then
83 self.editBox:SetHeight(math.max(height - 8, self:GetTextHeight()))
84 end
85 end)
86
87 -- Mouse handlers for focus
88 -- TRAP: Using OnMouseDown on parent frames intercepts the event required for
89 -- the native EditBox to start text selection/highlighting.
90 -- SOLUTION: Use OnMouseUp on the scrollFrame to allow clicking empty space to focus.
91 self.scrollFrame:SetScript("OnMouseUp", function()
92 if not self.editBox:HasFocus() then
93 self.editBox:SetFocus()
94 self.editBox:SetCursorPosition(self.editBox:GetNumLetters())
95 end
96 end)
97
98 -- Auto-scroll to bottom on text change (optional)
99 self.editBox:SetScript("OnTextChanged", function(eb, userInput)
100 -- Read-only enforcement: revert user changes but allow programmatic updates
101 if self.readOnly and userInput then
102 eb:SetText(self.currentText or "")
103 return
104 end
105
106 if not self.paused then
107 self.scrollFrame:SetVerticalScroll(self.scrollFrame:GetVerticalScrollRange())
108 end
109 end)
110
111 -- Tab behavior
112 self.editBox:SetScript("OnTabPressed", function(eb)
113 eb:Insert(" ")
114 end)
115
116 -- Escape to clear focus
117 self.editBox:SetScript("OnEscapePressed", function(eb)
118 eb:ClearFocus()
119 end)
120 end

148 function MultiLineEditBoxMixin:SetReadOnly(readOnly)
149 self.readOnly = readOnly
150
151 -- Ensure the edit box is always enabled so it can receive focus for selection/copying.
152 -- Native EditBox:SetEnabled(false) makes it non-selectable.
153 self.editBox:SetEnabled(true)
154
155 if readOnly then
156 -- Block character input (typing)
157 self.editBox:SetScript("OnChar", function() end)
158
159 -- Handle key presses to allow navigation/copy but block modification
160 self.editBox:SetScript("OnKeyDown", function(eb, key)
161 -- Allow modifiers (needed for Shift+Arrow selection, Ctrl+C, etc.)
162 if
163 key == "LSHIFT"
164 or key == "RSHIFT"
165 or key == "LCTRL"
166 or key == "RCTRL"
167 or key == "LALT"
168 or key == "RALT"
169 then
170 return
171 end
172
173 -- Allow Ctrl+C (Copy) and Ctrl+A (Select All)
174 if IsControlKeyDown() then
175 if key == "C" then
176 return
177 end
178 if key == "A" then
179 self:SelectAll()
180 return
181 end
182 end
183
184 -- Allow navigation keys
185 if
186 key == "LEFT"
187 or key == "RIGHT"
188 or key == "UP"
189 or key == "DOWN"
190 or key == "HOME"
191 or key == "END"
192 or key == "PAGEUP"
193 or key == "PAGEDOWN"
194 or key == "ESCAPE"
195 then
196 return
197 end
198
199 -- Block everything else (Backspace, Delete, Enter, etc.)
200 eb:SetPropagateKeyboardInput(false)
201 end)
202 else
203 -- Restore default behavior
204 self.editBox:SetScript("OnChar", nil)
205 self.editBox:SetScript("OnKeyDown", nil)
206 end
207 end

221 --- Create a multi-line edit box with scroll support
222 ---@param parent Frame Parent frame
223 ---@param config table Configuration
224 ---@return Frame editBox
225 function FenUI:CreateMultiLineEditBox(parent, config)
226 config = config or {}
227
228 -- We use a Layout as the container for border/background
229 local container = self:CreateLayout(parent, {
230 name = config.name,
231 border = config.border or "Inset",
232 background = config.background or "surfaceInset",
233 width = config.width or 400,
234 height = config.height or (config.numLines and (config.numLines * 14 + 10)) or 200,
235 })
236
237 FenUI.Mixin(container, MultiLineEditBoxMixin)
238 container:Init(config)
239
240 return container
241 end
