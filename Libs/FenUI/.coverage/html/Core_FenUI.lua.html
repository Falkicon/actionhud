26 -- Detect the addon path by looking at where this file was loaded from
27 -- Uses debugstack to find the actual path
28 local function DetectAddonPath()
29 local stack = debugstack(1, 1, 0)
30 -- debugstack returns formats like:
31 -- [string "Interface\AddOns\!Mechanic\Libs\FenUI\Core\FenUI.lua"]:45: in...
32 -- [string "@Interface\AddOns\!Mechanic\Libs\FenUI\Core\FenUI.lua"]:45: in...
33 -- Interface\AddOns\!Mechanic\Libs\FenUI\Core\FenUI.lua:45: in...
34
35 -- Try multiple patterns to extract the file path
36 local path = stack:match('"@?(.-)"') -- From [string "..."] or [string "@..."]
37 or stack:match("@([^:]+)") -- From @path:line format
38 or stack:match("^([^:]+)") -- From path:line format
39
40 if not path then
41 return "Interface\\AddOns\\FenUI"
42 end
43
44 -- Remove trailing line number if present (e.g., "FenUI.lua:45" -> "FenUI.lua")
45 path = path:gsub(":%d+$", "")
46
47 -- Normalize to backslashes for WoW pathing
48 path = path:gsub("/", "\\")
49
50 -- Find the directory containing the Core folder
51 -- e.g. "Interface\AddOns\!Mechanic\Libs\FenUI\Core\FenUI.lua" -> "Interface\AddOns\!Mechanic\Libs\FenUI"
52 local addonDir = path:match("(.-)\\Core\\FenUI%.lua$")
53 if addonDir and addonDir ~= "" then
54 return addonDir
55 end
56
57 -- Try without the $ anchor in case of extra characters
58 addonDir = path:match("(.-)\\Core\\FenUI%.lua")
59 if addonDir and addonDir ~= "" then
60 return addonDir
61 end
62
63 -- Last resort fallback
64 return "Interface\\AddOns\\FenUI"
65 end
66
67 FenUI.ADDON_PATH = DetectAddonPath()

97 -- Frame Registry (for global theme changes)
98 --------------------------------------------------------------------------------
99
100 FenUI.registeredFrames = {}
101 FenUI.frameCount = 0
102
103 --- Register a frame for global theme updates
104 ---@param frame Frame The frame to register
105 ---@param frameType string Type of frame ("panel", "tabs", "button", etc.)
106 ---@return number frameId Unique identifier for the frame
107 function FenUI:RegisterFrame(frame, frameType)
108 self.frameCount = self.frameCount + 1
109 local frameId = self.frameCount
110
111 self.registeredFrames[frameId] = {
112 frame = frame,
113 frameType = frameType or "unknown",
114 theme = nil, -- Will be set when theme is applied
115 }
116
117 -- Store the ID on the frame for easy lookup
118 frame.fenUIFrameId = frameId
119
120 self:Debug("Registered frame", frameId, "type:", frameType)
121 return frameId
122 end
123
124 --- Unregister a frame from global theme updates
125 ---@param frameOrId Frame|number The frame or its ID
126 function FenUI:UnregisterFrame(frameOrId)
127 local frameId = type(frameOrId) == "number" and frameOrId or frameOrId.fenUIFrameId
128
129 if frameId and self.registeredFrames[frameId] then
130 self.registeredFrames[frameId] = nil
131 self:Debug("Unregistered frame", frameId)
132 end
133 end
134
135 --- Get all registered frames
136 ---@return table<number, table> The registered frames
137 function FenUI:GetRegisteredFrames()
138 return self.registeredFrames
139 end

158 ---@param callbackId number

163 --- Fire theme change callbacks
164 ---@param themeName string
165 local function FireThemeChangeCallbacks(themeName)
166 for _, callback in pairs(FenUI.themeChangeCallbacks) do
167 local ok, err = pcall(callback, themeName)
168 if not ok then

210 SLASH_FENUI1 = "/fenui"
211 SlashCmdList["FENUI"] = function(msg)
212 local cmd, arg = strsplit(" ", strlower(msg or ""), 2)
213
214 if cmd == "validate" or cmd == "check" then
215 if FenUI.Validation then
216 local verbose = arg == "verbose"
217 local results = FenUI.Validation:Run(verbose)
218 FenUI.Validation:PrintReport(results, verbose)
219 else
220 FenUI:Print("Validation module not loaded")
221 end
222 elseif cmd == "theme" then
223 if arg then
224 FenUI:SetGlobalTheme(arg)
225 else
226 FenUI:Print("Current theme:", FenUI.currentGlobalTheme)
227 FenUI:Print("Available themes:", table.concat(FenUI:GetThemeList(), ", "))
228 end
229 elseif cmd == "themes" then
230 FenUI:Print("Available themes:")
231 for _, theme in ipairs(FenUI:GetThemeList()) do
232 local marker = (theme == FenUI.currentGlobalTheme) and " |cff00ff00(active)|r" or ""
233 FenUI:Print(" -", theme, marker)
234 end
235 elseif cmd == "debug" then
236 FenUI:SetDebugMode(not FenUI.debugMode)
237 FenUIDB.debugMode = FenUI.debugMode
238 elseif cmd == "version" then
239 FenUI:Print("Version", FenUI.VERSION, "by", FenUI.AUTHOR)
240 elseif cmd == "tokens" then
241 FenUI:Print("Current Tokens (Resolution):")
242 FenUI:Print(" marginPanel:", FenUI:GetSpacing("marginPanel"))
243 FenUI:Print(" panelPadding:", FenUI:GetLayout("panelPadding"))
244 FenUI:Print(" headerHeight:", FenUI:GetLayout("headerHeight"))
245 FenUI:Print(" footerHeight:", FenUI:GetLayout("footerHeight"))
246 elseif cmd == "frames" then
247 local count = 0
248 for _ in pairs(FenUI.registeredFrames) do
249 count = count + 1
250 end
251 FenUI:Print("Registered frames:", count)
252 if FenUI.debugMode then
253 for id, data in pairs(FenUI.registeredFrames) do
254 FenUI:Print(" -", id, "(", data.frameType, ")", "theme:", data.theme or "none")
255 end
256 end
257 else
258 FenUI:Print("FenUI v" .. FenUI.VERSION .. " Commands:")
259 FenUI:Print(" /fenui validate [verbose] - Check Blizzard dependencies")
260 FenUI:Print(" /fenui theme [name] - Get/set global theme")
261 FenUI:Print(" /fenui themes - List available themes")
262 FenUI:Print(" /fenui frames - Show registered frame count")
263 FenUI:Print(" /fenui tokens - Show current spacing tokens")
264 FenUI:Print(" /fenui debug - Toggle debug mode")
265 FenUI:Print(" /fenui version - Show version info")
266 end

273 --- Mixin: Copy methods from source tables to a target object
274 ---@param object table The target object
275 ---@param ... table One or more source tables
276 ---@return table The modified object
277 function FenUI.Mixin(object, ...)
278 for i = 1, select("#", ...) do
