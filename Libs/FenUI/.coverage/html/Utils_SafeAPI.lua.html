8 -- Internal cache for performance
9 local cooldownCache = {}
10 local chargesCache = {}
11 local textureCache = {}
12 local overlayCache = {}
13 local cacheTime = 0
14 local CACHE_DURATION = 0.016 -- ~1 frame
15
16 local function InvalidateCache()
17 local now = _G.GetTime()
18 if now - cacheTime > CACHE_DURATION then
19 _G.wipe(cooldownCache)
20 _G.wipe(chargesCache)
21 _G.wipe(overlayCache)
22 cacheTime = now
23 end
24 end
