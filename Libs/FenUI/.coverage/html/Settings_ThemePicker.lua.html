15 --- Create an AceConfig-compatible option table for theme selection
16 ---@param config table Configuration
17 ---@return table aceConfigOption
18 function FenUI:CreateThemeOption(config)
19 config = config or {}
20
21 local option = {
22 type = "select",
23 name = config.name or "UI Theme",
24 desc = config.desc or "Select the visual theme for FenUI components",
25 order = config.order or 1,
26 values = function()
27 local values = {}
28 for _, themeName in ipairs(FenUI:GetThemeList()) do
29 local theme = FenUI:GetTheme(themeName)
30 values[themeName] = theme and theme.name or themeName
31 end
32 return values
33 end,
34 get = config.get or function()
35 return FenUI:GetGlobalTheme()
36 end,
37 set = config.set or function(info, value)
38 FenUI:SetGlobalTheme(value)
39 end,
40 }
41
42 return option
43 end

129 function ThemePickerMixin:Init(config)
130 self.config = config or {}
131 self.savedVars = config.savedVars
132 self.savedVarKey = config.savedVarKey or "theme"
133 self.onChange = config.onChange
134
135 self:Refresh()
136 end
137
138 function ThemePickerMixin:Refresh()
139 -- Clear existing buttons
140 if self.buttons then
141 for _, btn in pairs(self.buttons) do
142 btn:Hide()
143 end
144 end
145 self.buttons = {}
146
147 -- Create theme buttons
148 local themes = FenUI:GetThemeList()
149 local currentTheme = self:GetCurrentTheme()
150
151 local yOffset = 0
152 for _, themeName in ipairs(themes) do
153 local theme = FenUI:GetTheme(themeName)
154 local btn = self:CreateThemeButton(themeName, theme, yOffset)
155 btn:SetSelected(themeName == currentTheme)
156 self.buttons[themeName] = btn
157 yOffset = yOffset - 28
158 end
159
160 -- Update container height
161 self:SetHeight(math.abs(yOffset) + 10)
162 end
163
164 function ThemePickerMixin:CreateThemeButton(themeName, theme, yOffset)
165 local btn = CreateFrame("Button", nil, self)
166 btn:SetHeight(24)
167 btn:SetPoint("TOPLEFT", 0, yOffset)
168 btn:SetPoint("TOPRIGHT", 0, yOffset)
169
170 btn.themeName = themeName
171
172 -- Background
173 btn.bg = btn:CreateTexture(nil, "BACKGROUND")
174 btn.bg:SetAllPoints()
175 btn.bg:SetColorTexture(0, 0, 0, 0)
176
177 -- Selection indicator
178 btn.selected = btn:CreateTexture(nil, "BORDER")
179 btn.selected:SetPoint("LEFT", 4, 0)
180 btn.selected:SetSize(4, 16)
181 btn.selected:SetColorTexture(FenUI:GetColor("interactiveSelected"))
182 btn.selected:Hide()
183
184 -- Text
185 btn.text = btn:CreateFontString(nil, "OVERLAY")
186 btn.text:SetFontObject(FenUI:GetFont("fontBody"))
187 btn.text:SetPoint("LEFT", 14, 0)
188 btn.text:SetText(theme and theme.name or themeName)
189 local r, g, b = FenUI:GetColor("textDefault")
190 btn.text:SetTextColor(r, g, b)
191
192 -- Description
193 if theme and theme.description then
194 btn.desc = btn:CreateFontString(nil, "OVERLAY")
195 btn.desc:SetFontObject(FenUI:GetFont("fontSmall"))
196 btn.desc:SetPoint("LEFT", btn.text, "RIGHT", 8, 0)
197 btn.desc:SetText("- " .. theme.description)
198 local mr, mg, mb = FenUI:GetColor("textMuted")
199 btn.desc:SetTextColor(mr, mg, mb)
200 end
201
202 function btn:SetSelected(selected)
203 self.isSelected = selected
204 self.selected:SetShown(selected)
205 if selected then
206 local sr, sg, sb = FenUI:GetColor("interactiveSelected")
207 self.text:SetTextColor(sr, sg, sb)
208 else
209 local tr, tg, tb = FenUI:GetColor("textDefault")
210 self.text:SetTextColor(tr, tg, tb)
211 end
212 end
213
214 -- Hover
215 btn:SetScript("OnEnter", function(self)
216 self.bg:SetColorTexture(FenUI:GetColor("surfaceElevated"))
217 if not self.isSelected then
218 local hr, hg, hb = FenUI:GetColor("interactiveHover")
219 self.text:SetTextColor(hr, hg, hb)
220 end
221 end)
222
223 btn:SetScript("OnLeave", function(self)
224 self.bg:SetColorTexture(0, 0, 0, 0)
225 if not self.isSelected then
226 local tr, tg, tb = FenUI:GetColor("textDefault")
227 self.text:SetTextColor(tr, tg, tb)
228 end
229 end)
230
231 -- Click
232 btn:SetScript("OnClick", function(self)
233 self:GetParent():SelectTheme(self.themeName)
234 end)
235
236 return btn
237 end

239 function ThemePickerMixin:GetCurrentTheme()
240 if self.savedVars and self.savedVarKey then
241 return self.savedVars[self.savedVarKey] or FenUI:GetGlobalTheme()
242 end
243 return FenUI:GetGlobalTheme()
244 end
245
246 function ThemePickerMixin:SelectTheme(themeName)
247 -- Update all buttons
248 for name, btn in pairs(self.buttons) do
249 btn:SetSelected(name == themeName)
250 end
251
252 -- Save to saved vars
253 if self.savedVars and self.savedVarKey then
254 self.savedVars[self.savedVarKey] = themeName
255 end
256
257 -- Set global theme
258 FenUI:SetGlobalTheme(themeName)
259
260 -- Fire callback
261 if self.onChange then
262 self.onChange(themeName)
263 end
264 end

266 --- Create a standalone theme picker widget
267 ---@param parent Frame Parent frame
268 ---@param savedVars table SavedVariables table to store selection
269 ---@param savedVarKey string Key in savedVars for the theme
270 ---@param onChange function Callback when theme changes
271 ---@return Frame themePicker
272 function FenUI:CreateThemePicker(parent, savedVars, savedVarKey, onChange)
273 local picker = CreateFrame("Frame", nil, parent)
274 FenUI.Mixin(picker, ThemePickerMixin)
275
276 picker:SetWidth(parent and parent:GetWidth() or 200)
277
278 picker:Init({
279 savedVars = savedVars,
280 savedVarKey = savedVarKey,
281 onChange = onChange,
282 })
283
284 return picker
285 end
