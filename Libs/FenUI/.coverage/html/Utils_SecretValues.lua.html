8 --- Check if a value is a Midnight secret value.
9 ---@param value any
10 ---@return boolean isSecret
11 function Utils:IsValueSecret(value)
12 if not self.IS_MIDNIGHT then
13 return false
14 end
15 if value == nil then
16 return false
17 end
18
19 -- Official checker
20 if _G.issecretvalue then
21 local ok, secret = pcall(_G.issecretvalue, value)
22 return ok and (secret == true)
23 end
24
25 -- Fallback: Check types that can be secret

28 return false
29 end
30
31 -- Robust fallback: pcall a blocked operation
32 local ok = pcall(function()
33 local _ = tostring(value)

37 end

39 --- Counts secret values within a table (shallow or recursive).
40 ---@param tbl table
41 ---@param recursive boolean?
42 ---@return number count
43 function Utils:CountSecrets(tbl, recursive)
