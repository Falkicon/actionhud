30 function WidgetMixin:Refresh()
31 -- Clear current rows
32 for _, row in ipairs(self.rows) do
33 row:Hide()
34 table.insert(self.rowPool, row)
35 end

38 local yOffset = 0
39 local function addNode(node, depth)
40 local row = self:GetRow()
41 row:SetPoint("TOPLEFT", depth * 16, -yOffset)
42 row:SetPoint("TOPRIGHT", 0, -yOffset)
43 row:SetText(node.text)
44 row.value = node.value
45 row:Show()
46
47 table.insert(self.rows, row)
48 yOffset = yOffset + 20
49
50 local expanded = node.expanded
51 if expanded == nil then
52 expanded = true
53 end -- Default to expanded
54
55 if node.children and expanded then
56 for _, child in ipairs(node.children) do
57 addNode(child, depth + 1)
58 end
59 end
60 end
61
62 if self.data then
63 for _, node in ipairs(self.data) do
64 addNode(node, 0)
65 end
66 end
67
68 self.content:SetHeight(yOffset)
69 end
