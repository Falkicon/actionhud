2 -- FenUI - Section Widget
3 --
4 -- A content primitive for structured information display.
5 -- Consists of a heading (bold) and body text (normal).
6 -- Used for help dialogs, about boxes, release notes, feature descriptions.
7 --------------------------------------------------------------------------------
8
9 local FenUI = FenUI
10 local SectionMixin = {}
11
12 function SectionMixin:Init(config)
13 config = config or {}
14 self.config = config
15
16 -- 1. Heading FontString
17 local heading = self:CreateFontString(nil, "OVERLAY", config.headingFont or "GameFontNormalLarge")
18 heading:SetText(config.heading or "")
19 heading:SetJustifyH("LEFT")
20 heading:SetWordWrap(false)
21
22 local hr, hg, hb = FenUI:GetColorRGB(config.headingColor or "textHeading")
23 heading:SetTextColor(hr, hg, hb)
24
25 heading:SetPoint("TOPLEFT", 0, 0)
26 heading:SetPoint("TOPRIGHT", 0, 0)
27 self.heading = heading
28
29 -- 2. Body FontString
30 local body = self:CreateFontString(nil, "OVERLAY", config.bodyFont or "GameFontHighlight")
31 body:SetText(config.body or "")
32 body:SetJustifyH("LEFT")
33 body:SetJustifyV("TOP")
34 body:SetWordWrap(true)
35 body:SetNonSpaceWrap(true)
36
37 local br, bg, bb = FenUI:GetColorRGB(config.bodyColor or "textDefault")
38 body:SetTextColor(br, bg, bb)
39
40 -- Spacing between heading and body
41 local gap = FenUI:GetSpacing(config.gap or 0)
42 body:SetPoint("TOPLEFT", heading, "BOTTOMLEFT", 0, -gap)
43 body:SetPoint("TOPRIGHT", heading, "BOTTOMRIGHT", 0, -gap)
44 self.body = body
45
46 -- 3. Calculate initial height
47 self:UpdateHeight()
48
49 -- 4. Handle resize (body text wrapping may change)
50 self:SetScript("OnSizeChanged", function()
51 self:UpdateHeight()
52 end)
53 end
54
55 function SectionMixin:UpdateHeight()
56 local headingHeight = self.heading:GetStringHeight()
57 local bodyHeight = self.body:GetStringHeight()
58 local gap = FenUI:GetSpacing(self.config.gap or 0)
59
60 self:SetHeight(headingHeight + gap + bodyHeight)
61 end
62
63 function SectionMixin:SetHeading(text)
64 self.heading:SetText(text)
65 self:UpdateHeight()
66 end
67
68 function SectionMixin:SetBody(text)
69 self.body:SetText(text)
70 self:UpdateHeight()
71 end
72
73 function SectionMixin:SetContent(heading, body)
74 self.heading:SetText(heading or "")
75 self.body:SetText(body or "")
76 self:UpdateHeight()
77 end
78
79 function SectionMixin:GetHeight()
80 return self.heading:GetStringHeight() + FenUI:GetSpacing(self.config.gap or 0) + self.body:GetStringHeight()
81 end

87 --- Create a section with heading and body text
88 ---@param parent Frame Parent frame
89 ---@param config table { heading, body, headingFont, bodyFont, headingColor, bodyColor, gap }
90 ---@return Frame section
91 function FenUI:CreateSection(parent, config)
92 local section = CreateFrame("Frame", nil, parent)
93 FenUI.Mixin(section, SectionMixin)
94 section:Init(config)
95 return section
