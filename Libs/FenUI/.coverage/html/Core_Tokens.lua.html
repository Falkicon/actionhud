266 --- Resolve a primitive color token to RGBA values
267 ---@param tokenName string The color token name (e.g., "gold500")
268 ---@return number, number, number, number r, g, b, a values
269 local function ResolvePrimitiveColor(tokenName)
270 local color = FenUI.Tokens.colors[tokenName]
271 if color then
272 return color[1], color[2], color[3], color[4] or 1
273 end
274 -- Fallback to white if token not found
275 FenUI:Debug("Unknown color token:", tokenName)
276 return 1, 1, 1, 1
277 end
278
279 --- Resolve a spacing token to a pixel value
280 ---@param tokenName string The spacing token name (e.g., "md")
281 ---@return number pixels
282 local function ResolvePrimitiveSpacing(tokenName)
283 local spacing = FenUI.Tokens.spacing[tokenName]
284 if spacing then
285 return spacing
286 end
287 FenUI:Debug("Unknown spacing token:", tokenName)
288 return 0
289 end
290
291 --- Resolve a font token to a font object name
292 ---@param tokenName string The font token name
293 ---@return string fontObjectName
294 local function ResolvePrimitiveFont(tokenName)
295 local font = FenUI.Tokens.fonts[tokenName]
296 if font then
297 return font
298 end
299 FenUI:Debug("Unknown font token:", tokenName)
300 return "GameFontNormal"
301 end

307 --- Get a color by semantic token name
308 --- Resolves through: overrides -> semantic -> primitive
309 ---@param semanticToken string The semantic token name (e.g., "surfacePanel")
310 ---@return number, number, number, number r, g, b, a values
311 function FenUI:GetColor(semanticToken)
312 -- Check for theme overrides first
313 local primitiveToken = self.Tokens.currentOverrides[semanticToken] or self.Tokens.semantic[semanticToken]
314
315 if primitiveToken then
316 return ResolvePrimitiveColor(primitiveToken)
317 end
318
319 -- If it's already a primitive token name, resolve directly
320 if self.Tokens.colors[semanticToken] then
321 return ResolvePrimitiveColor(semanticToken)
322 end
323
324 FenUI:Debug("Unknown semantic color token:", semanticToken)
325 return 1, 1, 1, 1
326 end

329 ---@param semanticToken string
330 ---@return table color {r, g, b, a}
331 function FenUI:GetColorTable(semanticToken)
332 local r, g, b, a = self:GetColor(semanticToken)

334 end

434 --- Apply token overrides (used by ThemeManager)
435 ---@param overrides table<string, string> Token overrides (semantic -> primitive)
436 function FenUI:ApplyTokenOverrides(overrides)
437 wipe(self.Tokens.currentOverrides)
438 if overrides then
439 for semantic, primitive in pairs(overrides) do
440 self.Tokens.currentOverrides[semantic] = primitive
441 end
442 end
443 FenUI:Debug("Applied token overrides:", overrides and #overrides or 0, "tokens")
444 end
445
446 --- Clear all token overrides
447 function FenUI:ClearTokenOverrides()
448 wipe(self.Tokens.currentOverrides)
449 FenUI:Debug("Cleared token overrides")
450 end

452 --- Get all semantic token names for a category
453 ---@param prefix string Token prefix (e.g., "surface", "text", "border")
454 ---@return table<string, string> tokens
455 function FenUI:GetTokensByPrefix(prefix)
456 local tokens = {}
457 for name, value in pairs(self.Tokens.semantic) do
458 if name:find("^" .. prefix) then
459 tokens[name] = value
460 end
461 end
462 return tokens
