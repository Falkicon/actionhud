7 local FenUI = FenUI
8 local WidgetMixin = {}
9
10 function WidgetMixin:Init(config)
11 self.config = config or {}
12
13 -- Background/Border using Layout
14 local layout = FenUI:CreateLayout(self, {
15 background = "surfaceInset",
16 border = "Inset",
17 padding = 4,
18 })
19 layout:SetAllPoints()
20 self.layout = layout
21
22 -- EditBox
23 local editBox = CreateFrame("EditBox", nil, self)
24 editBox:SetAllPoints(layout)
25 editBox:SetFontObject("ChatFontNormal")
26 editBox:SetAutoFocus(false)
27 editBox:SetTextInsets(8, 8, 0, 0)
28
29 if self.config.placeholder then
30 local placeholder = editBox:CreateFontString(nil, "OVERLAY", "ChatFontNormal")
31 placeholder:SetPoint("LEFT", 8, 0)
32
33 local U = FenUI.Utils
34 local placeholderText = U and U:SanitizeText(self.config.placeholder, "Enter text...")
35 or (self.config.placeholder == true and "Enter text..." or self.config.placeholder)
36
37 placeholder:SetText(placeholderText)
38 placeholder:SetTextColor(0.5, 0.5, 0.5)
39 self.placeholder = placeholder
40
41 editBox:SetScript("OnTextChanged", function(eb)
42 placeholder:SetShown(eb:GetText() == "")
43 if self.config.onChange then
44 self.config.onChange(eb:GetText())
45 end
46 end)
47 else
48 editBox:SetScript("OnTextChanged", function(eb)
49 if self.config.onChange then
50 self.config.onChange(eb:GetText())
51 end
52 end)
53 end
54
55 if self.config.readOnly then
56 editBox:SetEnabled(false)
57 end
58
59 -- Enter/Escape handlers
60 editBox:SetScript("OnEnterPressed", function(eb)
61 if self.config.onEnter then
62 self.config.onEnter(eb:GetText())
63 end
64 eb:ClearFocus()
65 end)
66
67 editBox:SetScript("OnEscapePressed", function(eb)
68 if self.config.onEscape then
69 self.config.onEscape(eb:GetText())
70 end
71 eb:ClearFocus()
72 end)
73
74 self.editBox = editBox
75 end
76
77 function WidgetMixin:SetText(text)
78 self.editBox:SetText(text or "")
79 end
80
81 function WidgetMixin:GetText()
82 return self.editBox:GetText()
83 end

85 -- Factory function
86 function FenUI:CreateInput(parent, config)
87 local frame = CreateFrame("Frame", nil, parent)
88 frame:SetSize(config.width or 200, config.height or 24)
89
90 FenUI.Mixin(frame, WidgetMixin)
91 frame:Init(config)
92
