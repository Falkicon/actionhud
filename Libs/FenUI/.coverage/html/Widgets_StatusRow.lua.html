13 local StatusRowMixin = {}

15 function StatusRowMixin:Init(config)
16 self.config = config or {}
17 self.items = {} -- { { label, value, labelFS, valueFS, divider }, ... }
18
19 self:SetHeight(config.height or 24)
20
21 if config.items then
22 self:SetValues(config.items)
23 end
24
25 self:SetScript("OnSizeChanged", function()
26 self:UpdateLayout()
27 end)
28 end
29
30 function StatusRowMixin:SetValues(valuesTable)
31 -- valuesTable = { { label, value }, ... }
32 self:Clear()
33
34 for i, item in ipairs(valuesTable) do
35 local entry = {}
36
37 -- Label
38 local labelFS = self:CreateFontString(nil, "OVERLAY", "GameFontNormalSmall")
39 labelFS:SetFormattedText("%s:", item.label)
40 labelFS:SetTextColor(FenUI:GetColorRGB("textMuted"))
41 entry.labelFS = labelFS
42
43 -- Value
44 local valueFS = self:CreateFontString(nil, "OVERLAY", "GameFontHighlightSmall")
45 valueFS:SetText(item.value)
46 entry.valueFS = valueFS
47
48 -- Divider (except for last item)
49 if i < #valuesTable then
50 local divider = self:CreateTexture(nil, "ARTWORK")
51 divider:SetColorTexture(FenUI:GetColorRGB("borderSubtle"))
52 divider:SetSize(1, self:GetHeight() * 0.6)
53 entry.divider = divider
54 end
55
56 entry.label = item.label
57 entry.value = item.value
58
59 table.insert(self.items, entry)
60 end
61
62 self:UpdateLayout()
63 end
64
65 function StatusRowMixin:SetValue(label, newValue)
66 for _, item in ipairs(self.items) do
67 if item.label == label then
68 item.value = newValue
69 item.valueFS:SetText(newValue)
70 self:UpdateLayout()
71 return
72 end
73 end
74 end
75
76 function StatusRowMixin:Clear()
77 for _, item in ipairs(self.items) do
78 item.labelFS:Hide()
79 item.valueFS:Hide()
80 if item.divider then
81 item.divider:Hide()
82 end
83 end
84 wipe(self.items)
85 end
86
87 function StatusRowMixin:UpdateLayout()
88 local xOffset = 8
89 local gap = self.config.gap or 12
90 local internalGap = 4
91
92 for i, item in ipairs(self.items) do
93 item.labelFS:ClearAllPoints()
94 item.labelFS:SetPoint("LEFT", self, "LEFT", xOffset, 0)
95 item.labelFS:Show()
96 xOffset = xOffset + item.labelFS:GetWidth() + internalGap
97
98 item.valueFS:ClearAllPoints()
99 item.valueFS:SetPoint("LEFT", self, "LEFT", xOffset, 0)
100 item.valueFS:Show()
101 xOffset = xOffset + item.valueFS:GetWidth()
102
103 if item.divider then
104 xOffset = xOffset + (gap / 2)
105 item.divider:ClearAllPoints()
106 item.divider:SetPoint("LEFT", self, "LEFT", xOffset, 0)
107 item.divider:Show()
108 xOffset = xOffset + (gap / 2)
109 end
110
111 xOffset = xOffset + gap
112 end
113 end
