16 function ToolbarMixin:Init(config)
17 config = config or {}
18 self.config = config
19 self.items = {}

22 self:SetWidth(config.width)
23 end
24 self:SetHeight(config.height or FenUI:GetLayout("buttonHeight") + 8)

84 function ToolbarMixin:UpdateLayout()
85 local width = self:GetWidth()
86 local gap = self.config.gap or FenUI:GetSpacing("spacingTight")
87 local padding = self.config.padding or { left = 0, right = 0 }
88 if type(padding) == "number" then
89 padding = { left = padding, right = padding }
90 end
91
92 local availableWidth = width - (padding.left or 0) - (padding.right or 0)
93
94 -- Pass 1: Calculate fixed width and count flex spacers
95 local fixedWidth = 0
96 local flexCount = 0
97 local visibleCount = 0
98
99 for _, item in ipairs(self.items) do
100 if item.type == "spacer" and item.width == "flex" then
101 flexCount = flexCount + 1
102 else
103 if item.frame then
104 fixedWidth = fixedWidth + item.frame:GetWidth()
105 elseif item.type == "spacer" then
106 fixedWidth = fixedWidth + (item.width or 0)
107 end
108 visibleCount = visibleCount + 1
109 end
110 end
111
112 -- Add gaps
113 fixedWidth = fixedWidth + (math.max(0, visibleCount + flexCount - 1) * gap)
114
115 local flexWidth = flexCount > 0 and (availableWidth - fixedWidth) / flexCount or 0
116
117 -- Pass 2: Position items
118 local xOffset = padding.left or 0
119 local align = self.config.align or "left"
120
121 -- Alignment adjustments
122 if align == "right" and flexCount == 0 then
123 xOffset = width - fixedWidth - (padding.right or 0)
124 elseif align == "center" and flexCount == 0 then
125 xOffset = (width - fixedWidth) / 2
126 end
127
128 for _, item in ipairs(self.items) do
129 if item.type == "spacer" and item.width == "flex" then
130 xOffset = xOffset + flexWidth + gap
131 else
132 local itemWidth = 0
133 if item.frame then
134 item.frame:ClearAllPoints()
135 item.frame:SetPoint("LEFT", self, "LEFT", xOffset, 0)
136 item.frame:Show()
137 itemWidth = item.frame:GetWidth()
138 elseif item.type == "spacer" then
139 itemWidth = item.width or 0
140 end
141 xOffset = xOffset + itemWidth + gap
142 end
143 end
144 end
