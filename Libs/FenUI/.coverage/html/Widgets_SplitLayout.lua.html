11 function SplitLayoutMixin:InitSplit(config)
12 self.config = config or {}
13
14 -- SplitLayout is a 2-column Layout
15 -- Cell 1: Navigation
16 -- Cell 2: Content
17
18 local navWidth = config.navWidth or 200
19
20 -- Content Area (Cell 2)
21 self.contentArea = self:GetCell(2)
22
23 -- Navigation Area (Cell 1)
24 local navCell = self:GetCell(1)
25 self.navPanel = navCell
26
27 -- Navigation Background
28 -- We use a dedicated texture on the navigation cell to provide visual depth.
29 -- FenUI:GetColor resolves the "surfaceInset" semantic token to its primitive RGBA values.
30 -- We then override the alpha (0.4) to create a semi-transparent "sunken" sidebar look
31 -- that matches modern Blizzard UI (e.g., Settings Panel) without needing an explicit border.
32 self.navBackground = navCell:CreateTexture(nil, "BACKGROUND")
33 self.navBackground:SetAllPoints()
34 local r, g, b, a = FenUI:GetColor("surfaceInset")
35 self.navBackground:SetColorTexture(r, g, b, 0.1)
36
37 -- Navigation Border (Right edge separator)
38 self.navSeparator = navCell:CreateTexture(nil, "BORDER")
39 self.navSeparator:SetPoint("TOPRIGHT")
40 self.navSeparator:SetPoint("BOTTOMRIGHT")
41 self.navSeparator:SetWidth(1)
42 self.navSeparator:SetColorTexture(FenUI:GetColorRGB("borderSubtle"))
43
44 -- Navigation Scroll Frame
45 local navScrollPanel = FenUI:CreateScrollPanel(navCell, {
46 padding = 2,
47 })
48 navScrollPanel:SetAllPoints()
49 self.navScrollPanel = navScrollPanel
50 self.navScroll = navScrollPanel.scrollFrame
51 self.navContent = navScrollPanel.scrollChild
52
53 -- Sync navContent width
54 self.navScroll:SetScript("OnSizeChanged", function(frame, width)
55 self.navContent:SetWidth(width)
56 self:RefreshNav()
57 end)
58
59 -- State
60 self.items = {}
61 self.buttons = {}
62 self.headers = {}
63 self.selectedKey = config.defaultKey
64 self.contentFrames = {}
65 self.onSelect = config.onSelect
66
67 -- Methods (Ensure they are on the instance)
68 self.SetItems = SplitLayoutMixin.SetItems
69 self.RefreshNav = SplitLayoutMixin.RefreshNav
70 self.GetOrCreateButton = SplitLayoutMixin.GetOrCreateButton
71 self.GetOrCreateHeader = SplitLayoutMixin.GetOrCreateHeader
72 self.UpdateButtonStates = SplitLayoutMixin.UpdateButtonStates
73 self.Select = SplitLayoutMixin.Select
74 self.GetContentFrame = SplitLayoutMixin.GetContentFrame
75 self.GetSelectedKey = SplitLayoutMixin.GetSelectedKey
76
77 if config.items then
78 self:SetItems(config.items)
79 end
80
81 -- Trigger initial selection if requested
82 if self.selectedKey then
83 self:Select(self.selectedKey, true)
84 end

88 self.items = items or {}

92 function SplitLayoutMixin:RefreshNav()
93 if not self.navContent then
94 return
95 end
96
97 -- Hide all existing buttons and headers
98 for _, btn in ipairs(self.buttons) do
99 btn:Hide()
100 end
101 for _, header in ipairs(self.headers) do
102 header:Hide()
103 end
104
105 local yOffset = 0
106 local buttonIndex = 1
107 local headerIndex = 1
108
109 for i, item in ipairs(self.items) do
110 if item.isHeader or item.isCategory then
111 local header = self:GetOrCreateHeader(headerIndex)
112 header:SetPoint("TOPLEFT", self.navContent, "TOPLEFT", 0, -yOffset)
113 header:SetPoint("TOPRIGHT", self.navContent, "TOPRIGHT", 0, -yOffset)
114 header:SetText(item.text)
115 header:Show()
116
117 yOffset = yOffset + header:GetHeight() + 2
118 headerIndex = headerIndex + 1
119 else
120 local btn = self:GetOrCreateButton(buttonIndex)
121 btn:SetPoint("TOPLEFT", self.navContent, "TOPLEFT", 0, -yOffset)
122 btn:SetPoint("TOPRIGHT", self.navContent, "TOPRIGHT", 0, -yOffset)
123 btn.text:SetText(item.text)
124 btn.key = item.key
125
126 -- Handle icons
127 if item.icon then
128 btn.icon:SetTexture(item.icon)
129 btn.icon:Show()
130 btn.text:SetPoint("LEFT", 32, 0)
131 else
132 btn.icon:Hide()
133 btn.text:SetPoint("LEFT", 8, 0)
134 end
135
136 btn:Show()
137 yOffset = yOffset + 24 + 2
138 buttonIndex = buttonIndex + 1
139 end
140 end
141
142 self.navContent:SetHeight(math.max(1, yOffset))
143 if self.navScrollPanel then
144 self.navScrollPanel:UpdateScrollBar()
145 end
146 self:UpdateButtonStates()
147
148 -- Ensure content frame for selected key is shown
149 if self.selectedKey and self.contentFrames[self.selectedKey] then
150 self.contentFrames[self.selectedKey]:Show()
151 end
152 end

154 function SplitLayoutMixin:GetOrCreateButton(index)
155 if self.buttons[index] then
156 return self.buttons[index]
157 end
158
159 local btn = CreateFrame("Button", nil, self.navContent)
160 btn:SetHeight(24)
161
162 -- Background highlight (selected)
163 local highlight = btn:CreateTexture(nil, "BACKGROUND")
164 highlight:SetAllPoints()
165 highlight:SetColorTexture(1, 1, 1, 0.1)
166 highlight:Hide()
167 btn.highlight = highlight
168
169 -- Hover highlight
170 local hover = btn:CreateTexture(nil, "HIGHLIGHT")
171 hover:SetAllPoints()
172 hover:SetColorTexture(1, 1, 1, 0.05)
173
174 -- Text
175 local text = btn:CreateFontString(nil, "OVERLAY", "GameFontNormal")
176 text:SetPoint("LEFT", 8, 0)
177 text:SetPoint("RIGHT", -8, 0)
178 text:SetJustifyH("LEFT")
179 btn.text = text
180
181 -- Icon
182 local icon = btn:CreateTexture(nil, "OVERLAY")
183 icon:SetSize(16, 16)
184 icon:SetPoint("LEFT", 8, 0)
185 btn.icon = icon
186
187 btn:SetScript("OnClick", function()
188 if btn.key then
189 self:Select(btn.key)
190 end
191 end)
192
193 self.buttons[index] = btn
194 return btn
195 end

211 function SplitLayoutMixin:UpdateButtonStates()
212 for _, btn in ipairs(self.buttons) do
213 if btn:IsShown() then
214 if btn.key == self.selectedKey then
215 btn.highlight:Show()
216 btn.text:SetTextColor(1, 1, 1)
217 else
218 btn.highlight:Hide()
219 btn.text:SetTextColor(1, 0.82, 0)
220 end
221 end
222 end
223 end

225 function SplitLayoutMixin:Select(key, force)
226 if self.selectedKey == key and not force then
227 return
228 end
229
230 self.selectedKey = key
231 self:UpdateButtonStates()
232
233 -- Hide all content frames
234 for _, frame in pairs(self.contentFrames) do
235 frame:Hide()
236 end
237
238 -- Show selected content frame
239 if self.contentFrames[key] then
240 self.contentFrames[key]:Show()
241 end
242
243 -- Callback
244 if self.onSelect then
245 self.onSelect(key)
246 end
247 end
248
249 function SplitLayoutMixin:GetContentFrame(key)
250 if not self.contentFrames[key] then
251 local frame = CreateFrame("Frame", nil, self.contentArea)
252 frame:SetAllPoints()
253 self.contentFrames[key] = frame
254
255 -- If this is already the selected key, show it immediately
256 if self.selectedKey == key then
257 frame:Show()
258 else
259 frame:Hide()
260 end
261 end
262 return self.contentFrames[key]
263 end
264
265 function SplitLayoutMixin:GetSelectedKey()
266 return self.selectedKey
267 end

273 function FenUI:CreateSplitLayout(parent, config)
274 config = config or {}
275
276 -- SplitLayout is a 2-column Layout
277 local layout = self:CreateLayout(parent, {
278 name = config.name,
279 cols = { config.navWidth or 200, "fr" },
280 gap = config.gap or 4,
281 padding = config.padding or 0,
282 border = config.border,
283 background = config.background or "surfacePanel",
284 })
285
286 FenUI.Mixin(layout, SplitLayoutMixin)
287 layout:InitSplit(config)
288
