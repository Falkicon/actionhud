23 --- Initialize the panel with configuration
24 function PanelMixin:Init(config)
25 self.config = config or {}
26 self.slots = {}
27 self.hooks = {
28 onCreate = config.onCreate,
29 onShow = config.onShow,
30 onHide = config.onHide,
31 onThemeChange = config.onThemeChange,
32 }
33
34 -- Mark as supporting layouts for theme system
35 self.fenUISupportsLayout = true
36
37 -- Size is handled by Layout component or factory, but apply if needed
38 if not self.config.usesLayout then
39 self:SetSize(config.width or 400, config.height or 300)
40 end
41
42 -- Create SafeZone frame for systematic anchoring
43 -- This frame is inset to clear the thick Blizzard metal borders
44 self:CreateSafeZone()
45
46 -- Default padding (from SafeZone edges to content)
47 self.padding = { left = 0, right = 0, top = 0, bottom = 0 }
48 self:SetPadding(config.padding)
49
50 if config.title then
51 self:SetTitle(config.title)
52 end
53
54 if config.movable then
55 self:MakeMovable()
56 end
57
58 if config.resizable then
59 self:MakeResizable()
60 end
61
62 if config.closable ~= false then
63 self:CreateCloseButton()
64 end
65
66 -- Apply initial slots
67 if config.slots then
68 for slotName, frame in pairs(config.slots) do
69 self:SetSlot(slotName, frame)
70 end
71 end
72
73 -- Register for theme updates if requested
74 if config.registerForThemeChanges ~= false then
75 self:RegisterForThemeChanges()
76 end
77
78 -- Apply theme
79 local themeName = config.theme or FenUI:GetGlobalTheme()
80 FenUI:ApplyTheme(self, themeName)
81
82 -- Fire onCreate hook
83 if self.hooks.onCreate then
84 self.hooks.onCreate(self)
85 end
86 end

95 function PanelMixin:SetTitle(text)

200 function PanelMixin:CreateSafeZone()
201 if self.safeZone then
202 return
203 end
204
205 -- The SafeZone is a logical frame that represents the "safe" usable area
206 -- clear of borders and title bars.
207 self.safeZone = CreateFrame("Frame", nil, self)
208
209 -- Dynamic insets based on border pack and title presence
210 local borderOffset = self.contentInset or 4 -- Fallback if not using custom border yet
211
212 local left = borderOffset
213 local right = borderOffset
214 local top = borderOffset
215 local bottom = borderOffset
216
217 -- If we have a title bar, the safe zone top should clear it
218 if self.config.title then
219 top = top + 24 -- Space for title text
220 end
221
222 self.safeZone:SetPoint("TOPLEFT", self, "TOPLEFT", left, -top)
223 self.safeZone:SetPoint("BOTTOMRIGHT", self, "BOTTOMRIGHT", -right, bottom)
224 end
225
226 function PanelMixin:GetSafeZone()
227 if not self.safeZone then
228 self:CreateSafeZone()
229 end
230 return self.safeZone

261 --[[
262 Available slots:
263 - headerLeft: Left side of header (before title)
264 - headerRight: Right side of header (before close button)
265 - content: Main content area
266 - footerLeft: Left side of footer
267 - footerRight: Right side of footer
268 - footer: Entire footer area

270
271 function PanelMixin:SetSlot(slotName, frame)
272 if not frame then
273 return
274 end
275
276 -- Store the slot
277 self.slots[slotName] = frame
278
279 -- Parent and position the frame
280 frame:SetParent(self)
281
282 -- NOTE: Systematic Slot Positioning via SafeZone
283 -- We anchor slots to the SafeZone frame rather than the main frame.
284 -- This ensures they are automatically clear of the Blizzard metal border textures.
285 local safeZone = self.safeZone
286 local headerH = FenUI:GetLayout("headerHeight")
287 local footerH = FenUI:GetLayout("footerHeight")
288
289 if slotName == "headerLeft" then
290 frame:SetPoint("TOPLEFT", safeZone, "TOPLEFT", 0, 0)
291 elseif slotName == "headerRight" then
292 local offset = self.closeButton and -28 or 0
293 frame:SetPoint("TOPRIGHT", safeZone, "TOPRIGHT", offset, 0)
294 elseif slotName == "content" then
295 frame:SetPoint("TOPLEFT", safeZone, "TOPLEFT", 0, -headerH)
296 frame:SetPoint("BOTTOMRIGHT", safeZone, "BOTTOMRIGHT", 0, footerH)
297 elseif slotName == "footerLeft" then
298 frame:SetPoint("BOTTOMLEFT", safeZone, "BOTTOMLEFT", 0, 0)
299 elseif slotName == "footerRight" then
300 frame:SetPoint("BOTTOMRIGHT", safeZone, "BOTTOMRIGHT", 0, 0)
301 elseif slotName == "footer" then
302 frame:SetPoint("BOTTOMLEFT", safeZone, "BOTTOMLEFT", 0, 0)
303 frame:SetPoint("BOTTOMRIGHT", safeZone, "BOTTOMRIGHT", 0, 0)
304 end
305
306 frame:Show()
307 end

334 function PanelMixin:SetPadding(padding)
335 if type(padding) == "number" then
336 self.padding = { left = padding, right = padding, top = padding, bottom = padding }
337 elseif type(padding) == "table" then
338 self.padding.left = padding.left or self.padding.left
339 self.padding.right = padding.right or self.padding.right
340 self.padding.top = padding.top or self.padding.top
341 self.padding.bottom = padding.bottom or self.padding.bottom
342 end

344 -- Apply individual side overrides from config
345 local config = self.config
346 if config.paddingTop then
347 self.padding.top = FenUI:GetSpacing(config.paddingTop)
348 end
349 if config.paddingBottom then
350 self.padding.bottom = FenUI:GetSpacing(config.paddingBottom)
351 end
352 if config.paddingLeft then
353 self.padding.left = FenUI:GetSpacing(config.paddingLeft)
354 end
355 if config.paddingRight then

364 function PanelMixin:UpdateContentAnchors()
365 if not self.contentFrame then
366 return
367 end
368
369 local safeZone = self.safeZone
370 local headerH = FenUI:GetLayout("headerHeight")
371 local footerH = FenUI:GetLayout("footerHeight")
372
373 local p = self.padding
374 self.contentFrame:ClearAllPoints()
375 self.contentFrame:SetPoint("TOPLEFT", safeZone, "TOPLEFT", p.left, -(headerH + p.top))
376 self.contentFrame:SetPoint("BOTTOMRIGHT", safeZone, "BOTTOMRIGHT", -p.right, footerH + p.bottom)
377 end

428 --- Create a panel with configuration object
429 ---@param parent Frame Parent frame
430 ---@param config table|string Configuration table or just a title string
431 ---@return Frame panel
432 function FenUI:CreatePanel(parent, config)
433 -- Allow simple string as title
434 if type(config) == "string" then
435 config = { title = config }
436 end
437 config = config or {}
438
439 -- Determine layout/border
440 local theme = FenUI:GetTheme(config.theme)
441 local borderKey = config.layout or (theme and theme.layout) or "ModernDark"
442 local textureKit = config.textureKit or (theme and theme.textureKit)
443
444 -- Create base panel using Layout component
445 local panel
446 if FenUI.CreateLayout then
447 -- Use Layout as base (preferred)
448 local bgConfig = (config.background == nil) and "surfacePanel" or config.background
449 panel = FenUI:CreateLayout(parent or UIParent, {
450 name = config.name,
451 width = config.width or 400,
452 height = config.height or 300,
453 border = borderKey,
454 background = bgConfig,
455 shadow = config.shadow,
456 padding = config.padding,
457 paddingTop = config.paddingTop,
458 paddingBottom = config.paddingBottom,
459 paddingLeft = config.paddingLeft,
460 paddingRight = config.paddingRight,
461 margin = config.margin,
462 marginTop = config.marginTop,
463 marginBottom = config.marginBottom,
464 marginLeft = config.marginLeft,
465 marginRight = config.marginRight,
466 textureKit = textureKit,
467 minWidth = config.minWidth,
468 maxWidth = config.maxWidth,
469 minHeight = config.minHeight,
470 maxHeight = config.maxHeight,
471 aspectRatio = config.aspectRatio,
472 aspectBase = config.aspectBase,
473 resizable = config.resizable,
474 })
475 else
476 -- Fallback to direct frame creation
477 panel = CreateFrame("Frame", config.name, parent or UIParent)
478 panel:SetSize(config.width or 400, config.height or 300)
479 FenUI:ApplyBorder(panel, borderKey)
480 end
481
482 -- Apply Panel mixin (title, close button, slots, hooks)
483 FenUI.Mixin(panel, PanelMixin)
484
485 -- Initialize with config
486 panel:Init(config)
487
488 -- Apply constraints on size change
489 panel:HookScript("OnSizeChanged", function()
490 panel:ApplyConstraints()
491 end)
492
493 -- Set up show/hide hooks
494 panel:HookScript("OnShow", function(self)
495 if self.hooks.onShow then
496 self.hooks.onShow(self)
497 end
498 end)
499
500 panel:HookScript("OnHide", function(self)
501 if self.hooks.onHide then
502 self.hooks.onHide(self)
503 end
504 end)
505
506 return panel
507 end
508
509 --- Apply size constraints to the panel
510 function PanelMixin:ApplyConstraints()
511 local width = self:GetWidth()
512 local height = self:GetHeight()
513 local config = self.config
514
515 if config.minWidth and width < config.minWidth then
516 self:SetWidth(config.minWidth)
517 end
518 if config.minHeight and height < config.minHeight then
519 self:SetHeight(config.minHeight)
520 end
521 if config.maxWidth and width > config.maxWidth then
522 self:SetWidth(config.maxWidth)
523 end
524 if config.maxHeight and height > config.maxHeight then
525 self:SetHeight(config.maxHeight)
526 end
527
528 if self.contentFrame then
529 self:UpdateContentAnchors()
530 end
531 end
