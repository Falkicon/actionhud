22 local function GetSpacing(val)
23 if not val then
24 return 0
25 end
26 if type(val) == "string" then
27 return FenUI:GetSpacing(val)
28 elseif type(val) == "number" then
29 return val
30 end
31 return 0

64 --- Create an inset container (styled content area)
65 ---@param parent Frame Parent frame
66 ---@param config table|nil Configuration { padding, alpha, background, shadow }
67 ---@return Frame inset
68 function FenUI:CreateInset(parent, config)
69 config = config or {}
70
71 local inset
72
73 -- Use Layout component if available (preferred)
74 if FenUI.CreateLayout then
75 -- Determine background config
76 local bgConfig = config.background
77 if not bgConfig then
78 -- Default to surfaceInset with alpha
79 if config.alpha then
80 bgConfig = { color = "surfaceInset", alpha = config.alpha }
81 else
82 bgConfig = "surfaceInset"
83 end
84 end
85
86 inset = FenUI:CreateLayout(parent, {
87 border = "Inset",
88 background = bgConfig,
89 shadow = config.shadow,
90 })
91 else
92 -- Fallback to original implementation
93 inset = CreateFrame("Frame", nil, parent, "BackdropTemplate")
94 FenUI.Mixin(inset, InsetMixin)
95 inset:Init(config)
96 end
97
98 -- Position using layout constants or config
99 -- NOTE: Systematic Margin Application
100 -- By default, insets are positioned with 0 padding unless specified.
101 local padding = GetSpacing(config.padding or 0)
102 local topOffset = config.topOffset or 0
103 local bottomOffset = config.bottomOffset or 0
104
105 inset:ClearAllPoints()
106 inset:SetPoint("TOPLEFT", parent, "TOPLEFT", padding, -topOffset)
107 inset:SetPoint("BOTTOMRIGHT", parent, "BOTTOMRIGHT", -padding, bottomOffset)
108
109 -- Add convenience method for backwards compatibility
110 if not inset.SetInsetAlpha then
111 function inset:SetInsetAlpha(alpha)
112 if self.bgTexture then
113 local r, g, b = FenUI:GetColorRGB("surfaceInset")
114 self.bgTexture:SetColorTexture(r, g, b, alpha)
115 end
116 end
117 end
118
119 return inset
120 end

146 function ScrollPanelMixin:ScrollToTop()
147 self.scrollFrame:SetVerticalScroll(0)
148 end
149
150 function ScrollPanelMixin:ScrollToBottom()
151 local maxScroll = self.scrollFrame:GetVerticalScrollRange()
152 self.scrollFrame:SetVerticalScroll(maxScroll)
153 end
154
155 function ScrollPanelMixin:UpdateScrollBar()
156 if not self.scrollBar then
157 return
158 end
159
160 local visibleHeight = self.scrollFrame:GetHeight()
161 local totalHeight = self.scrollChild:GetHeight()
162
163 -- If height is 0, we might be in the first frame.
164 -- Retry in next frame to ensure layout is done.
165 if visibleHeight <= 0 or totalHeight <= 0 then
166 if not self.initRetry then
167 self.initRetry = true
168 C_Timer.After(0.1, function()
169 self.initRetry = false
170 self:UpdateScrollBar()
171 end)
172 end
173 return
174 end
175
176 self.scrollBar:UpdateThumbSize(visibleHeight, totalHeight)
177 end

179 --- Create a scroll panel (scrollable content area)
180 ---@param parent Frame Parent frame
181 ---@param config table|nil Configuration { padding, showScrollBar }
182 ---@return Frame scrollPanel
183 function FenUI:CreateScrollPanel(parent, config)
184 config = config or {}
185
186 local container = CreateFrame("Frame", nil, parent)
187 FenUI.Mixin(container, ScrollPanelMixin)
188
189 local padding = GetSpacing(config.padding or 0)
190 local scrollBarWidth = config.showScrollBar ~= false and GetLayout("scrollBarWidth") or 0
191
192 -- 1. Create native ScrollFrame (no template)
193 local scrollFrame = CreateFrame("ScrollFrame", nil, container)
194 scrollFrame:SetPoint("TOPLEFT", padding, -padding)
195 scrollFrame:SetPoint("BOTTOMRIGHT", -(padding + scrollBarWidth), padding)
196 container.scrollFrame = scrollFrame
197
198 -- 2. Create custom ScrollBar
199 if config.showScrollBar ~= false then
200 local scrollBar = self:CreateScrollBar(container, {
201 width = scrollBarWidth,
202 })
203 scrollBar:SetPoint("TOPRIGHT", -padding, -padding)
204 scrollBar:SetPoint("BOTTOMRIGHT", -padding, padding)
205
206 -- Link scrollBar and scrollFrame
207 scrollBar:SetScript("OnValueChanged", function(_, value)
208 scrollFrame:SetVerticalScroll(value)
209 end)
210
211 scrollFrame:SetScript("OnScrollRangeChanged", function(_, xrange, yrange)
212 container:UpdateScrollBar()
213 end)
214
215 scrollFrame:SetScript("OnVerticalScroll", function(_, offset)
216 scrollBar:SetValue(offset)
217 end)
218
219 -- Mouse wheel support
220 scrollFrame:EnableMouseWheel(true)
221 scrollFrame:SetScript("OnMouseWheel", function(_, delta)
222 local current = scrollBar:GetValue()
223 scrollBar:SetValue(current - (delta * 20)) -- 20px per scroll step
224 end)
225
226 container.scrollBar = scrollBar
227 end
228
229 -- 3. Create scroll child
230 local scrollChild = CreateFrame("Frame", nil, scrollFrame)
231 scrollChild:SetWidth(1) -- Set by OnSizeChanged
232 scrollChild:SetHeight(1) -- Will be set by content
233
234 -- INSPECT SUPPORT: Enable picking for the content area
235 scrollChild:EnableMouse(true)
236 if scrollChild.SetMouseClickEnabled then
237 scrollChild:SetMouseClickEnabled(false)
238 end
239
240 scrollFrame:SetScrollChild(scrollChild)
241 container.scrollChild = scrollChild
242
243 -- Update scroll child width and scrollbar when container resizes
244 container:SetScript("OnSizeChanged", function(self, width, height)
245 local innerWidth = width - (padding * 2) - scrollBarWidth
246 scrollChild:SetWidth(math.max(1, innerWidth))
247 self:UpdateScrollBar()
248 end)
249
250 container:Init(config)
251
252 return container
253 end

259 --- Create an inset container with built-in scroll functionality
260 ---@param parent Frame Parent frame
261 ---@param config table|nil Configuration { padding, topOffset, bottomOffset, alpha, scrollPadding }
262 ---@return Frame inset, Frame scrollChild
263 function FenUI:CreateScrollInset(parent, config)
264 config = config or {}
265
266 -- Create the inset container
267 local inset = self:CreateInset(parent, config)
268
269 -- Create scroll panel inside it
270 local scrollPanel = self:CreateScrollPanel(inset, {
271 padding = config.scrollPadding or 0,
272 showScrollBar = config.showScrollBar ~= false,
273 })
274 scrollPanel:SetAllPoints()
275
276 -- Attach scroll panel to inset for easy access
277 inset.scrollPanel = scrollPanel
278 inset.scrollChild = scrollPanel.scrollChild
279
280 -- Convenience methods
281 function inset:GetScrollChild()
282 return self.scrollChild
283 end
284
285 function inset:SetContentHeight(height)
286 self.scrollPanel:SetContentHeight(height)
287 end
288
289 function inset:ScrollToTop()
290 self.scrollPanel:ScrollToTop()
291 end
292
293 return inset, scrollPanel.scrollChild
294 end
